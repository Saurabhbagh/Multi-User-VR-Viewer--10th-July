420  public Anchor FindPreferredAnchor() {
421  if (!useTrajectory) {
422  // Simply try to attach to the nearest valid anchor.
423  return GetNearestValidAnchor();
424  }
425  else {
426  // Pick the nearby valid anchor with the highest score, based on proximity and trajectory.
427  Anchor optimalAnchor = null;
428  float optimalScore = 0F;
429  Anchor testAnchor = null;
430  float testScore = 0F;
431  foreach (var anchor in GetNearbyValidAnchors()) {
432  testAnchor = anchor;
433  testScore = getAnchorScore(anchor);
434 
435  // Scores of 0 mark ineligible anchors.
436  if (testScore == 0F) continue;
437 
438  if (testScore > optimalScore) {
439  optimalAnchor = testAnchor;
440  optimalScore = testScore;
441  }
442  }
443 
444  return optimalAnchor;
445  }
446  }

492  public Anchor GetNearestValidAnchor(bool requireWithinRange = true,
493  bool requireAnchorHasSpace = true,
494  bool requireAnchorActiveAndEnabled = true) {
495  HashSet<Anchor> anchorsToCheck;
496 
497  if (this.anchorGroup == null) {
498  anchorsToCheck = Anchor.allAnchors;
499  }
500  else {
501  anchorsToCheck = this.anchorGroup.anchors;
502  }
503 
504  Anchor closestAnchor = null;
505  float closestDistSqrd = float.PositiveInfinity;
506  foreach (var testAnchor in anchorsToCheck) {
507  if (requireAnchorHasSpace) {
508  bool anchorHasSpace = testAnchor.anchoredObjects.Count == 0
509  || testAnchor.allowMultipleObjects;
510  if (!anchorHasSpace) {
511  // Skip the anchor for consideration.
512  continue;
513  }
514  }
515  if (requireAnchorActiveAndEnabled && !testAnchor.isActiveAndEnabled) {
516  // Skip the anchor for consideration.
517  continue;
518  }
519 
520  float testDistanceSqrd = (testAnchor.transform.position - this.transform.position).sqrMagnitude;
521  if (testDistanceSqrd < closestDistSqrd) {
522  closestAnchor = testAnchor;
523  closestDistSqrd = testDistanceSqrd;
524  }
525  }
526 
527  if (!requireWithinRange || closestDistSqrd < maxAnchorRange * maxAnchorRange) {
528  return closestAnchor;
529  }
530  else {
531  return null;
532  }

586  public static float GetAnchorScore(Vector3 anchObjPos, Vector3 anchObjVel, Vector3 anchorPos, float maxDistance, float nonDirectedMaxDistance, float minAngleProduct,
587  float alwaysAttachDistance = 0f) {
588  // Calculated a "directedness" heuristic for determining whether the user is throwing or releasing without directed motion.
589  float directedness = anchObjVel.magnitude.Map(0.20F, 1F, 0F, 1F);
590 
591  float effMaxDistance = directedness.Map(0F, 1F, nonDirectedMaxDistance, maxDistance);
592  Vector3 effPos = Utils.Map(Mathf.Sqrt(Mathf.Sqrt(directedness)), 0f, 1f,
593  anchObjPos, (anchObjPos - anchObjVel.normalized * effMaxDistance * 0.30f));
594 
595  float distanceSqrd = (anchorPos - effPos).sqrMagnitude;
596  float distanceScore;
597  if (distanceSqrd > effMaxDistance * effMaxDistance) {
598  distanceScore = 0F;
599  }
600  else {
601  distanceScore = distanceSqrd.Map(0F, effMaxDistance * effMaxDistance, 1F, 0F);
602  }
603 
604  float angleScore;
605  float dotProduct = Vector3.Dot(anchObjVel.normalized, (anchorPos - effPos).normalized);
606 
607  // Angular score only factors in based on how directed the motion of the object is.
608  dotProduct = Mathf.Lerp(1F, dotProduct, directedness);
609 
610  angleScore = dotProduct.Map(minAngleProduct, 1f, 0f, 1f);
611  angleScore *= angleScore;
612 
613  // Support an "always-attach distance" within which only distanceScore matters
614  float semiDistanceSqrd = (anchorPos - Vector3.Lerp(anchObjPos, effPos, 0.5f)).sqrMagnitude;
615  float useAlwaysAttachDistanceAmount = semiDistanceSqrd.Map(0f, Mathf.Max(0.0001f, (0.25f * alwaysAttachDistance * alwaysAttachDistance)),
616  1f, 0f);
617  angleScore = useAlwaysAttachDistanceAmount.Map(0f, 1f, angleScore, 1f);
618 
619  return distanceScore * angleScore;
620  }

622  private void updateAttractionToHand() {
623  if (interactionBehaviour == null || !isAttractedByHand) {
624  if (_offsetTowardsHand != Vector3.zero) {
625  _offsetTowardsHand = Vector3.Lerp(_offsetTowardsHand, Vector3.zero, 5F * Time.deltaTime);
626  }
627 
628  return;
629  }

631  float reachTargetAmount = 0F;
632  Vector3 towardsHand = Vector3.zero;
633  if (interactionBehaviour.isHovered) {
634  Hand hoveringHand = interactionBehaviour.closestHoveringHand;
635 
636  reachTargetAmount = Mathf.Clamp01(attractionReachByDistance.Evaluate(
637  Vector3.Distance(hoveringHand.PalmPosition.ToVector3(), anchor.transform.position)
638  ));
639  towardsHand = hoveringHand.PalmPosition.ToVector3() - anchor.transform.position;
640  }
641 
642  Vector3 targetOffsetTowardsHand = towardsHand * maxAttractionReach * reachTargetAmount;
643  _offsetTowardsHand = Vector3.Lerp(_offsetTowardsHand, targetOffsetTowardsHand, 5 * Time.deltaTime);

646  private void updateAnchorAttachment() {
647  // Initialize position.
648  Vector3 finalPosition;
649  if (interactionBehaviour != null) {
650  finalPosition = interactionBehaviour.rigidbody.position;
651  }
652  else {
653  finalPosition = this.transform.position;
654  }
655 
656  // Update position based on anchor state.
657  Vector3 targetPosition = anchor.transform.position;
658  if (lockToAnchor) {
659  // In this state, we are simply locked directly to the anchor.
660  finalPosition = targetPosition + _offsetTowardsHand;
661 
662  // Reset anchor position storage; it can't be updated from this state.
663  _hasTargetPositionLastUpdate = false;
664  }
665  else if (lockWhenAttached) {
666  if (_isLockedToAnchor) {
667  // In this state, we are already attached to the anchor.
668  finalPosition = targetPosition + _offsetTowardsHand;
669 
670  // Reset anchor position storage; it can't be updated from this state.
671  _hasTargetPositionLastUpdate = false;
672  }
673  else {
674  // Undo any "reach towards hand" offset.
675  finalPosition -= _offsetTowardsHand;
676 
677  // If desired, automatically correct for the anchor itself moving while attempting to return to it.
678  if (matchAnchorMotionWhileAttaching && this.transform.parent != anchor.transform) {
679  if (_hasTargetPositionLastUpdate) {
680  finalPosition += (targetPosition - _targetPositionLastUpdate);
681  }
682 
683  _targetPositionLastUpdate = targetPosition;
684  _hasTargetPositionLastUpdate = true;
685  }
686 
687  // Lerp towards the anchor.
688  finalPosition = Vector3.Lerp(finalPosition, targetPosition, anchorLerpCoeffPerSec * Time.deltaTime);
689  if (Vector3.Distance(finalPosition, targetPosition) < 0.001F) {
690  _isLockedToAnchor = true;
691  }
692 
693  // Redo any "reach toward hand" offset.
694  finalPosition += _offsetTowardsHand;
695  }
696  }
697 
698  // Set final position.
699  if (interactionBehaviour != null) {
700  interactionBehaviour.rigidbody.position = finalPosition;
701  this.transform.position = finalPosition;
702  }
703  else {
704  this.transform.position = finalPosition;
705  }
706  }

708  private void updateAnchorAttachmentRotation() {
709  // Initialize rotation.
710  Quaternion finalRotation;
711  if (interactionBehaviour != null) {
712  finalRotation = interactionBehaviour.rigidbody.rotation;
713  }
714  else {
715  finalRotation = this.transform.rotation;
716  }
717 
718  // Update rotation based on anchor state.
719  Quaternion targetRotation = anchor.transform.rotation;
720  if (lockToAnchor) {
721  // In this state, we are simply locked directly to the anchor.
722  finalRotation = targetRotation;
723 
724  // Reset anchor rotation storage; it can't be updated from this state.
725  _hasTargetPositionLastUpdate = false;
726  }
727  else if (lockWhenAttached) {
728  if (_isRotationLockedToAnchor) {
729  // In this state, we are already attached to the anchor.
730  finalRotation = targetRotation;
731 
732  // Reset anchor rotation storage; it can't be updated from this state.
733  _hasTargetRotationLastUpdate = false;
734  }
735  else {
736  // If desired, automatically correct for the anchor itself moving while attempting to return to it.
737  if (matchAnchorMotionWhileAttaching && this.transform.parent != anchor.transform) {
738  if (_hasTargetRotationLastUpdate) {
739  finalRotation = (Quaternion.Inverse(_targetRotationLastUpdate) * targetRotation) * finalRotation;
740  }
741 
742  _targetRotationLastUpdate = targetRotation;
743  _hasTargetRotationLastUpdate = true;
744  }
745 
746  // Slerp towards the anchor rotation.
747  finalRotation = Quaternion.Slerp(finalRotation, targetRotation, anchorLerpCoeffPerSec * 0.8F * Time.deltaTime);
748 
749  if (Quaternion.Angle(targetRotation, finalRotation) < 2F) {
750  _isRotationLockedToAnchor = true;
751  }
752  }
753  }
754 
755  // Set final rotation.
756  if (interactionBehaviour != null) {
757  interactionBehaviour.rigidbody.rotation = finalRotation;
758  this.transform.rotation = finalRotation;
759  }
760  else {
761  this.transform.rotation = finalRotation;
762  }
