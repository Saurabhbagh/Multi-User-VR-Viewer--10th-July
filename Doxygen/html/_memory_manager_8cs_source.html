49  [MonoPInvokeCallback(typeof(Allocate))]
50  public static IntPtr Pin(UInt32 size, eLeapAllocatorType typeHint, IntPtr state) {
51  try {
52  //Construct a key to identify the desired allocation
53  PoolKey key = new PoolKey() {
54  type = typeHint,
55  size = size
56  };
57 
58  //Attempt to find the pool that holds this type of allocation
59  Queue<object> pool;
60  if (!_pooledMemory.TryGetValue(key, out pool)) {
61  //Construct a new pool if none exists yet
62  pool = new Queue<object>();
63  _pooledMemory[key] = pool;
64  }
65 
66  //Attempt to get an object from the pool
67  object memory;
68  if (EnablePooling && pool.Count > MinPoolSize) {
69  memory = pool.Dequeue();
70  } else {
71  //If the pool is empty, we need to construct a new object
72  switch (typeHint) {
73  default:
74  case eLeapAllocatorType.eLeapAllocatorType_Uint8:
75  memory = new byte[size];
76  break;
77  case eLeapAllocatorType.eLeapAllocatorType_Float:
78  memory = new float[(size + sizeof(float) - 1) / sizeof(float)];
79  break;
80  }
81  }
82 
83  //Pin the object so its address will not change
84  GCHandle handle = GCHandle.Alloc(memory, GCHandleType.Pinned);
85  IntPtr ptr = handle.AddrOfPinnedObject();
86 
87  //Put the information about the newly pinned allocation into the
88  //active memory map so it can be retrieved and freed layer.
89  _activeMemory.Add(ptr, new ActiveMemoryInfo() {
90  handle = handle,
91  key = key
92  });
93 
94  return ptr;
95  } catch (Exception e) {
96  UnityEngine.Debug.LogException(e);
97  }
