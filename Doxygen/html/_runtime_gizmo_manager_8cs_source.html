251  private void generateMeshes() {
252  _cubeMesh = new Mesh();
253  _cubeMesh.name = "RuntimeGizmoCube";
254  _cubeMesh.hideFlags = HideFlags.HideAndDontSave;
255 
256  List<Vector3> verts = new List<Vector3>();
257  List<int> indexes = new List<int>();
258 
259  Vector3[] faces = new Vector3[] { Vector3.forward, Vector3.right, Vector3.up };
260  for (int i = 0; i < 3; i++) {
261  addQuad(verts, indexes, faces[(i + 0) % 3], -faces[(i + 1) % 3], faces[(i + 2) % 3]);
262  addQuad(verts, indexes, -faces[(i + 0) % 3], faces[(i + 1) % 3], faces[(i + 2) % 3]);
263  }
264 
265  _cubeMesh.SetVertices(verts);
266  _cubeMesh.SetIndices(indexes.ToArray(), MeshTopology.Quads, 0);
267  _cubeMesh.RecalculateNormals();
268  _cubeMesh.RecalculateBounds();
269  _cubeMesh.UploadMeshData(true);
270 
271  _wireCubeMesh = new Mesh();
272  _wireCubeMesh.name = "RuntimeWireCubeMesh";
273  _wireCubeMesh.hideFlags = HideFlags.HideAndDontSave;
274 
275  verts.Clear();
276  indexes.Clear();
277 
278  for (int dx = 1; dx >= -1; dx -= 2) {
279  for (int dy = 1; dy >= -1; dy -= 2) {
280  for (int dz = 1; dz >= -1; dz -= 2) {
281  verts.Add(0.5f * new Vector3(dx, dy, dz));
282  }
283  }
284  }
285 
286  addCorner(indexes, 0, 1, 2, 4);
287  addCorner(indexes, 3, 1, 2, 7);
288  addCorner(indexes, 5, 1, 4, 7);
289  addCorner(indexes, 6, 2, 4, 7);
290 
291  _wireCubeMesh.SetVertices(verts);
292  _wireCubeMesh.SetIndices(indexes.ToArray(), MeshTopology.Lines, 0);
293  _wireCubeMesh.RecalculateBounds();
294  _wireCubeMesh.UploadMeshData(true);
295 
296  _wireSphereMesh = new Mesh();
297  _wireSphereMesh.name = "RuntimeWireSphereMesh";
298  _wireSphereMesh.hideFlags = HideFlags.HideAndDontSave;
299 
300  verts.Clear();
301  indexes.Clear();
302 
303  int totalVerts = CIRCLE_RESOLUTION * 3;
304  for (int i = 0; i < CIRCLE_RESOLUTION; i++) {
305  float angle = Mathf.PI * 2 * i / CIRCLE_RESOLUTION;
306  float dx = 0.5f * Mathf.Cos(angle);
307  float dy = 0.5f * Mathf.Sin(angle);
308 
309  for (int j = 0; j < 3; j++) {
310  indexes.Add((i * 3 + j + 0) % totalVerts);
311  indexes.Add((i * 3 + j + 3) % totalVerts);
312  }
313 
314  verts.Add(new Vector3(dx, dy, 0));
315  verts.Add(new Vector3(0, dx, dy));
316  verts.Add(new Vector3(dx, 0, dy));
317  }
318 
319  _wireSphereMesh.SetVertices(verts);
320  _wireSphereMesh.SetIndices(indexes.ToArray(), MeshTopology.Lines, 0);
321  _wireSphereMesh.RecalculateBounds();
322  _wireSphereMesh.UploadMeshData(true);
323  }

326  indexes.Add(verts.Count + 0);
327  indexes.Add(verts.Count + 1);
328  indexes.Add(verts.Count + 2);
329  indexes.Add(verts.Count + 3);

331  verts.Add(0.5f * (normal + axis1 + axis2));

334  verts.Add(0.5f * (normal - axis1 + axis2));

615  bool traverseHierarchy = true,

627  RelativeTo(collider.transform);
628

631  DrawCollider(collider, skipMatrixSetup: true);

637  public void DrawCollider(Collider collider, bool useWireframe = true,
638  bool skipMatrixSetup = false) {
639  if (!skipMatrixSetup) {
640  PushMatrix();
641  RelativeTo(collider.transform);
642  }
643 
644  if (collider is BoxCollider) {
645  BoxCollider box = collider as BoxCollider;
646  if (useWireframe) {
647  DrawWireCube(box.center, box.size);
648  }
649  else {
650  DrawCube(box.center, box.size);
651  }
652  }
653  else if (collider is SphereCollider) {
654  SphereCollider sphere = collider as SphereCollider;
655  if (useWireframe) {
656  DrawWireSphere(sphere.center, sphere.radius);
657  }
658  else {
659  DrawSphere(sphere.center, sphere.radius);
660  }
661  }
662  else if (collider is CapsuleCollider) {
663  CapsuleCollider capsule = collider as CapsuleCollider;
664  if (useWireframe) {
665  Vector3 capsuleDir;
666  switch (capsule.direction) {
667  case 0: capsuleDir = Vector3.right; break;
668  case 1: capsuleDir = Vector3.up; break;
669  case 2: default: capsuleDir = Vector3.forward; break;
670  }
671  DrawWireCapsule(capsule.center + capsuleDir * (capsule.height / 2F - capsule.radius),
672  capsule.center - capsuleDir * (capsule.height / 2F - capsule.radius), capsule.radius);
673  }
674  else {
675  Vector3 size = Vector3.zero;
676  size += Vector3.one * capsule.radius * 2;
677  size += new Vector3(capsule.direction == 0 ? 1 : 0,
678  capsule.direction == 1 ? 1 : 0,
679  capsule.direction == 2 ? 1 : 0) * (capsule.height - capsule.radius * 2);
680  DrawCube(capsule.center, size);
681  }
682  }
683  else if (collider is MeshCollider) {
684  MeshCollider mesh = collider as MeshCollider;
685  if (mesh.sharedMesh != null) {
686  if (useWireframe) {
687  DrawWireMesh(mesh.sharedMesh, Matrix4x4.identity);
688  }
689  else {
690  DrawMesh(mesh.sharedMesh, Matrix4x4.identity);
691  }
692  }
693  }
694 
695  if (!skipMatrixSetup) {
696  PopMatrix();
697  }
698  }

710  public void DrawPosition(Vector3 pos, Color lerpColor, float lerpCoeff, float? overrideScale = null) {
711  float targetScale;
712  if (overrideScale.HasValue) {
713  targetScale = overrideScale.Value;
714  }
715  else {

717 
718  var curCam = Camera.current;
719  var posWorldSpace = matrix * pos;
720  if (curCam != null) {
721  float camDistance = Vector3.Distance(posWorldSpace, curCam.transform.position);
722 
723  targetScale *= camDistance;
724  }
725  }
726

728 
729  float negativeAlpha = 0.6f;

742  if (lerpCoeff != 0f) { color = color.LerpHSV(lerpColor, lerpCoeff); }
743  DrawLine(pos, pos - Vector3.up * extent);

745  color = Color.blue;

747  DrawLine(pos, pos + Vector3.forward * extent);

803 
804  for (int i = 0; i < _operations.Count; i++) {
805  OperationType type = _operations[i];

807  case OperationType.SetMatrix:
808  _currMatrix = _matrices[matrixIndex++];
809  break;
810 
811  case OperationType.SetColor:
812  _currColor = _colors[colorIndex++];
813  currPass = -1; //force pass to be set the next time we need to draw
814  break;

818  break;

822 
823  GL.Begin(GL.LINES);
824  Line line = _lines[lineIndex++];
825  GL.Vertex(_currMatrix.MultiplyPoint(line.a));
826  GL.Vertex(_currMatrix.MultiplyPoint(line.b));
827  GL.End();
828  break;
829 
830  case OperationType.DrawWireSphere:
831  setPass(ref currPass, isUnlit: true);
832 
833  GL.Begin(GL.LINES);
834  WireSphere wireSphere = _wireSpheres[wireSphereIndex++];
835  drawWireSphereNow(wireSphere, ref currPass);

837  break;
838 
839  case OperationType.DrawMesh:
840  if (GL.wireframe) {
841  setPass(ref currPass, isUnlit: true);
842  } else {
843  setPass(ref currPass, isUnlit: false);
844  }
845 
846  Graphics.DrawMeshNow(_meshes[meshIndex++],
847  _currMatrix * _matrices[matrixIndex++]);
848  break;
849 
850  default:
851  throw new InvalidOperationException("Unexpected operation type " + type);
852  }
853  }
854  } finally {
855  GL.wireframe = false;
856  }

904  var nextR = Q * r;

906  var nextOnPlaneAngle = Infix.Infix.SignedAngle(nextR, onPlaneDir, circleNormal);
