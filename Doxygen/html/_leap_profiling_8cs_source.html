20  public static class LeapProfiling {
21 
22  //Maps a block name to the sampler for it
23  private static Dictionary<string, CustomSampler> _samplers = new Dictionary<string, CustomSampler>();
24 
25  //Represents a queue of samplers that need to be created.
26  //Samplers can only be created on the main thread, so we need some way for the info given
27  //on alternate threads to be passed to the main thread.
28  private static Queue<string> _samplersToCreate = new Queue<string>();
29 
30  //We keep track of the size of the queue in a member variable
31  private static int _samplersToCreateCount = 0;
32 
33  public static void Update() {
34  //Read of _samplersToCreateCount is atomic
35  if (_samplersToCreateCount > 0) {
36  //Only if the count is nonzero do we do an expensive lock
37  lock (_samplersToCreate) {
38  //First duplicate the existing dictionary
39  var newDictionary = new Dictionary<string, CustomSampler>(_samplers);
40 
41  //Then construct all of the new samplers and add them to the new dictionary
42  while (_samplersToCreate.Count > 0) {
43  string blockName = _samplersToCreate.Dequeue();
44 
45  newDictionary[blockName] = CustomSampler.Create(blockName);
46  }
47 
48  //Reset samplers to create to zero
49  _samplersToCreateCount = 0;
50 
51  //Reference assignments are atomic in C#
52  //All new callbacks will now reference the updated dictionary
53  //Old dictionary will be collected by GC
54  _samplers = newDictionary;
55  }
56  }
57  }

59  public static void BeginProfilingForThread(BeginProfilingForThreadArgs eventData) {
60 #if UNITY_2017_3_OR_NEWER
61  //Enable unity profiling for this thread
62  Profiler.BeginThreadProfiling("LeapCSharp", eventData.threadName);
63 
64  //Assume that threads are not stopping and starting frequently
65  //so we can get away with less-than-optimal strategies when starting a thread.
66  //in this case we use a naive queue with a lock.
67  lock (_samplersToCreate) {
68  foreach (var blockName in eventData.blockNames) {
69  _samplersToCreate.Enqueue(blockName);
70  }
71 
72  Interlocked.Add(ref _samplersToCreateCount, eventData.blockNames.Length);
73  }
74 #else
75  Debug.LogWarning("Thread Profiling is unavailable in versions of Unity below 2017.3");
76 #endif
77  }

87  public static void BeginProfilingBlock(BeginProfilingBlockArgs eventData) {
88  //Sampler might not have been created yet because samplers can only be created
89  //on the main thread. We will simply not be able to report all blocks until
90  //a sampler is available.
91 
92  //Note that the Dictionary type is thread safe for read operations
93  //Dictionary is only used once and so there is no risk of the dictionary
94  //being swapped out from underneath us
95 
96  CustomSampler sampler;
97  if (_samplers.TryGetValue(eventData.blockName, out sampler)) {
98  sampler.Begin();
99  }
100  }
101 
102  public static void EndProfilingBlock(EndProfilingBlockArgs eventData) {
103  CustomSampler sampler;
104  if (_samplers.TryGetValue(eventData.blockName, out sampler)) {
105  sampler.End();
106  }
