404  public void FinishPreInitialize()
405  {
406  List<SteamVR_Action> allActionsList = new List<SteamVR_Action>();
407  List<ISteamVR_Action_In> nonVisualInActionsList = new List<ISteamVR_Action_In>();
408  List<ISteamVR_Action_In> visualActionsList = new List<ISteamVR_Action_In>();
409  List<SteamVR_Action_Pose> poseActionsList = new List<SteamVR_Action_Pose>();
410  List<SteamVR_Action_Skeleton> skeletonActionsList = new List<SteamVR_Action_Skeleton>();
411  List<ISteamVR_Action_Out> outActionList = new List<ISteamVR_Action_Out>();
412 
413  if (SteamVR_Input.actions == null)
414  {
415  Debug.LogError("<b>[SteamVR Input]</b> Actions not initialized!");
416  return;
417  }
418 
419  for (int actionIndex = 0; actionIndex < SteamVR_Input.actions.Length; actionIndex++)
420  {
421  SteamVR_Action action = SteamVR_Input.actions[actionIndex];
422 
423  if (action.actionSet.GetActionSetData() == this)
424  {
425  allActionsList.Add(action);
426 
427  if (action is ISteamVR_Action_Boolean || action is ISteamVR_Action_Single || action is ISteamVR_Action_Vector2 || action is ISteamVR_Action_Vector3)
428  {
429  nonVisualInActionsList.Add((ISteamVR_Action_In)action);
430  }
431  else if (action is SteamVR_Action_Pose)
432  {
433  visualActionsList.Add((ISteamVR_Action_In)action);
434  poseActionsList.Add((SteamVR_Action_Pose)action);
435  }
436  else if (action is SteamVR_Action_Skeleton)
437  {
438  visualActionsList.Add((ISteamVR_Action_In)action);
439  skeletonActionsList.Add((SteamVR_Action_Skeleton)action);
440  }
441  else if (action is ISteamVR_Action_Out)
442  {
443  outActionList.Add((ISteamVR_Action_Out)action);
444  }
445  else
446  {
447  Debug.LogError("<b>[SteamVR Input]</b> Action doesn't implement known interface: " + action.fullPath);
448  }
449  }
450  }
451 
452  allActions = allActionsList.ToArray();
453  nonVisualInActions = nonVisualInActionsList.ToArray();
454  visualActions = visualActionsList.ToArray();
455  poseActions = poseActionsList.ToArray();
456  skeletonActions = skeletonActionsList.ToArray();
457  outActionArray = outActionList.ToArray();
458  }
