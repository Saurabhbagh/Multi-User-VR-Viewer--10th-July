102 
103  // Align bow
104  // Time lerp value used for ramping into drawn bow orientation
105  float lerp = Util.RemapNumberClamped( Time.time, nockLerpStartTime, ( nockLerpStartTime + lerpDuration ), 0f, 1f );
106 
107  float pullLerp = Util.RemapNumberClamped( nockToarrowHand.magnitude, minPull, maxPull, 0f, 1f ); // Normalized current state of bow draw 0 - 1
108 
109  Vector3 arrowNockTransformToHeadset = ( ( Player.instance.hmdTransform.position + ( Vector3.down * 0.05f ) ) - arrowHand.arrowNockTransform.parent.position ).normalized;
110  Vector3 arrowHandPosition = ( arrowHand.arrowNockTransform.parent.position + ( ( arrowNockTransformToHeadset * drawOffset ) * pullLerp ) ); // Use this line to lerp arrowHand nock position
111  //Vector3 arrowHandPosition = arrowHand.arrowNockTransform.position; // Use this line if we don't want to lerp arrowHand nock position
112 
113  Vector3 pivotToString = ( arrowHandPosition - pivotTransform.position ).normalized;
114  Vector3 pivotToLowerHandle = ( handleTransform.position - pivotTransform.position ).normalized;
115  bowLeftVector = -Vector3.Cross( pivotToLowerHandle, pivotToString );
116  pivotTransform.rotation = Quaternion.Lerp( nockLerpStartRotation, Quaternion.LookRotation( pivotToString, bowLeftVector ), lerp );
117 
118  // Move nock position
119  if ( Vector3.Dot( nockToarrowHand, -nockTransform.forward ) > 0 )
120  {
121  float distanceToarrowHand = nockToarrowHand.magnitude * lerp;
122 
123  nockTransform.localPosition = new Vector3( 0f, 0f, Mathf.Clamp( -distanceToarrowHand, -maxPull, 0f ) );
124 
125  nockDistanceTravelled = -nockTransform.localPosition.z;
126 
127  arrowVelocity = Util.RemapNumber( nockDistanceTravelled, minPull, maxPull, arrowMinVelocity, arrowMaxVelocity );
128 
129  drawTension = Util.RemapNumberClamped( nockDistanceTravelled, 0, maxPull, 0f, 1f );
130

132 
133  if ( nockDistanceTravelled > minPull )

136  }
137  else
138  {
139  pulled = false;
140  }
141 
142  if ( ( nockDistanceTravelled > ( lastTickDistance + hapticDistanceThreshold ) ) || nockDistanceTravelled < ( lastTickDistance - hapticDistanceThreshold ) )
143  {
144  ushort hapticStrength = (ushort)Util.RemapNumber( nockDistanceTravelled, 0, maxPull, bowPullPulseStrengthLow, bowPullPulseStrengthHigh );
145  hand.TriggerHapticPulse( hapticStrength );
146  hand.otherHand.TriggerHapticPulse( hapticStrength );

261  private void EvaluateHandedness()
262  {
263  var handType = hand.handType;
264 
265  if ( handType == SteamVR_Input_Sources.LeftHand )// Bow hand is further left than arrow hand.
266  {
267  // We were considering a switch, but the current controller orientation matches our currently assigned handedness, so no longer consider a switch
268  if ( possibleHandSwitch && currentHandGuess == Handedness.Left )
269  {
270  possibleHandSwitch = false;
271  }
272 
273  // If we previously thought the bow was right-handed, and were not already considering switching, start considering a switch
274  if ( !possibleHandSwitch && currentHandGuess == Handedness.Right )
275  {
276  possibleHandSwitch = true;
277  timeOfPossibleHandSwitch = Time.time;
278  }
279 
280  // If we are considering a handedness switch, and it's been this way long enough, switch
281  if ( possibleHandSwitch && Time.time > ( timeOfPossibleHandSwitch + timeBeforeConfirmingHandSwitch ) )
282  {
283  currentHandGuess = Handedness.Left;
284  possibleHandSwitch = false;
285  }
286  }
287  else // Bow hand is further right than arrow hand
288  {
289  // We were considering a switch, but the current controller orientation matches our currently assigned handedness, so no longer consider a switch
290  if ( possibleHandSwitch && currentHandGuess == Handedness.Right )
291  {
292  possibleHandSwitch = false;
293  }
294 
295  // If we previously thought the bow was right-handed, and were not already considering switching, start considering a switch
296  if ( !possibleHandSwitch && currentHandGuess == Handedness.Left )
297  {
298  possibleHandSwitch = true;
299  timeOfPossibleHandSwitch = Time.time;
300  }
301 
302  // If we are considering a handedness switch, and it's been this way long enough, switch
303  if ( possibleHandSwitch && Time.time > ( timeOfPossibleHandSwitch + timeBeforeConfirmingHandSwitch ) )
304  {
305  currentHandGuess = Handedness.Right;
306  possibleHandSwitch = false;
307  }
308  }
