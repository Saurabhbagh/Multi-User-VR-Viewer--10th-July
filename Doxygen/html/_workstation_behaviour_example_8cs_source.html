68  public WorkstationState workstationState {

138  private void onGraspedMovement(Vector3 preSolvePos, Quaternion preSolveRot,
139  Vector3 curPos, Quaternion curRot,
140  List<InteractionController> controllers) {
141  // If the workstation is not fully open when grasped, close it.
142  if (workstationState == WorkstationState.Opening) {
143  DeactivateWorkstation();
144  }
145 
146  // If the velocity of the object while grasped is too large, exit workstation mode.
147  if (_intObj.rigidbody.velocity.magnitude > MAX_SPEED_AS_WORKSTATION
148  || (_intObj.rigidbody.isKinematic && ((preSolvePos - curPos).magnitude / Time.fixedDeltaTime) > MAX_SPEED_AS_WORKSTATION)) {
149  DeactivateWorkstation();
150  }
151 
152  // Lock our upward axis while workstation mode is open.
153  if (workstationState == WorkstationState.Open) {
154  _intObj.rigidbody.rotation = (Quaternion.FromToRotation(_intObj.rigidbody.rotation * Vector3.up, Vector3.up)) * _intObj.rigidbody.rotation;
155  _intObj.transform.rotation = _intObj.rigidbody.rotation;
156  }
157  }
158 
159  private void onPostObjectGraspEnd() {
160  if (_anchObj.preferredAnchor == null) {
161  // Choose a good position and rotation for workstation mode and begin traveling there.
162  Vector3 targetPosition;
163 
164  // Choose the current position if our velocity is small.
165  if (_intObj.rigidbody.velocity.magnitude < MIN_SPEED_TO_ACTIVATE_TRAVELING) {
166  targetPosition = _intObj.rigidbody.position;
167  }
168  else {
169  targetPosition = determineWorkstationPosition();
170  }
171 
172  Quaternion targetRotation = determineWorkstationRotation(targetPosition);
173 
174  beginTraveling(_intObj.rigidbody.position, _intObj.rigidbody.velocity,
175  _intObj.rigidbody.rotation, _intObj.rigidbody.angularVelocity,
176  targetPosition, targetRotation);
177  }
178  else {
179  // Ensure the workstation is not active or being deactivated if
180  // we are attaching to an anchor.
181  DeactivateWorkstation();
182  }
183  }

204  private void beginTraveling(Vector3 initPosition, Vector3 initVelocity,
205  Quaternion initRotation, Vector3 initAngVelocity,
206  Vector3 targetPosition, Quaternion targetRotation) {
207  _initTravelTime = Time.time;
208  _initTravelPosition = initPosition;
209  _initTravelVelocity = initVelocity;
210  _initTravelRotation = initRotation;
211  _initTravelAngVelocity = initAngVelocity;
212 
213  float velMagnitude = _initTravelVelocity.magnitude;
214  if (velMagnitude > MAX_TRAVEL_SPEED) {
215  float capSpeedMultiplier = MAX_TRAVEL_SPEED / velMagnitude;
216  _initTravelVelocity *= capSpeedMultiplier;
217  }
218 
219  _effGravity = Vector3.Lerp(Vector3.zero, Physics.gravity, initVelocity.magnitude.Map(0.80F, 3F, 0F, 0.70F));
220 
221 
222 
223  _travelTargetPosition = targetPosition;
224  _travelTargetRotation = targetRotation;
225 
226  // Construct a single-use Tween that will last a specific duration
227  // and specify a custom callback as it progresses to update the
228  // object's position and rotation.
229  _travelTween = Tween.Single()
230  .OverTime(_initTravelVelocity.magnitude.Map(_minMaxTravelTimeFromThrowSpeed.x, _minMaxTravelTimeFromThrowSpeed.y,
231  _minMaxWorkstationTravelTime.x, _minMaxWorkstationTravelTime.y))
232  .OnProgress(onTravelTweenProgress)
233  .OnReachEnd(ActivateWorkstation) // When the tween is finished, open workstation mode.
234  .Play();
235  }
236 
237  private void onTravelTweenProgress(float progress) {
238  float curTime = Time.time;
239  Vector3 extrapolatedPosition = evaluatePosition(_initTravelPosition, _initTravelVelocity, _effGravity, _initTravelTime, curTime);
240  Quaternion extrapolatedRotation = evaluateRotation(_initTravelRotation, _initTravelAngVelocity, _initTravelTime, curTime);
241 
242  // Interpolate from the position and rotation that the object would naturally have over time
243  // (by movement due to inertia and by acceleration due to gravity)
244  // to the target position and rotation over the lifetime of the tween.
245  _intObj.rigidbody.position = Vector3.Lerp(extrapolatedPosition, _travelTargetPosition, progress);
246  _intObj.rigidbody.rotation = Quaternion.Slerp(extrapolatedRotation, _travelTargetRotation, progress);
247  }

257  private Vector3 evaluatePosition(Vector3 initialPosition, Vector3 initialVelocity, Vector3 gravity, float initialTime, float timeToEvaluate) {
258  float t = timeToEvaluate - initialTime;
259  return initialPosition + (initialVelocity * t) + (0.5f * gravity * t * t);
260  }
261 
265  private Quaternion evaluateRotation(Quaternion initialRotation, Vector3 angularVelocity, float initialTime, float timeToEvaluate) {
266  float t = timeToEvaluate - initialTime;
267  return Quaternion.Euler(angularVelocity * t) * initialRotation;
268  }

313  public static Vector3 DefaultDetermineWorkstationPosition(Vector3 userEyePosition, Quaternion userEyeRotation,
314  Vector3 workstationObjInitPosition, Vector3 workstationObjInitVelocity, float workstationObjRadius,
315  List<Vector3> otherWorkstationPositions, List<float> otherWorkstationRadii) {
316  // Push velocity away from the camera if necessary.
317  Vector3 towardsCamera = (userEyePosition - workstationObjInitPosition).normalized;
318  float towardsCameraness = Mathf.Clamp01(Vector3.Dot(towardsCamera, workstationObjInitVelocity.normalized));
319  workstationObjInitVelocity = workstationObjInitVelocity + Vector3.Lerp(Vector3.zero, -towardsCamera * 2.00F, towardsCameraness);
320 
321  // Calculate velocity direction on the XZ plane.
322  Vector3 groundPlaneVelocity = Vector3.ProjectOnPlane(workstationObjInitVelocity, Vector3.up);
323  float groundPlaneDirectedness = groundPlaneVelocity.magnitude.Map(0.003F, 0.40F, 0F, 1F);
324  Vector3 groundPlaneDirection = groundPlaneVelocity.normalized;
325 
326  // Calculate camera "forward" direction on the XZ plane.
327  Vector3 cameraGroundPlaneForward = Vector3.ProjectOnPlane(userEyeRotation * Vector3.forward, Vector3.up);
328  float cameraGroundPlaneDirectedness = cameraGroundPlaneForward.magnitude.Map(0.001F, 0.01F, 0F, 1F);
329  Vector3 alternateCameraDirection = (userEyeRotation * Vector3.forward).y > 0F ? userEyeRotation * Vector3.down : userEyeRotation * Vector3.up;
330  cameraGroundPlaneForward = Vector3.Slerp(alternateCameraDirection, cameraGroundPlaneForward, cameraGroundPlaneDirectedness);
331  cameraGroundPlaneForward = cameraGroundPlaneForward.normalized;
332 
333  // Calculate a placement direction based on the camera and throw directions on the XZ plane.
334  Vector3 placementDirection = Vector3.Slerp(cameraGroundPlaneForward, groundPlaneDirection, groundPlaneDirectedness);
335 
336  // Calculate a placement position along the placement direction between min and max placement distances.
337  float minPlacementDistance = 0.25F;
338  float maxPlacementDistance = 0.51F;
339  Vector3 placementPosition = userEyePosition + placementDirection * Mathf.Lerp(minPlacementDistance, maxPlacementDistance,
340  (groundPlaneDirectedness * workstationObjInitVelocity.magnitude)
341  .Map(0F, 1.50F, 0F, 1F));
342 
343  // Don't move far if the initial velocity is small.
344  float overallDirectedness = workstationObjInitVelocity.magnitude.Map(0.00F, 3.00F, 0F, 1F);
345  placementPosition = Vector3.Lerp(workstationObjInitPosition, placementPosition, overallDirectedness * overallDirectedness);
346 
347  // Enforce placement height.
348  float placementHeightFromCamera = -0.30F;
349  placementPosition.y = userEyePosition.y + placementHeightFromCamera;
350 
351  // Enforce minimum placement away from user.
352  Vector2 cameraXZ = new Vector2(userEyePosition.x, userEyePosition.z);
353  Vector2 stationXZ = new Vector2(placementPosition.x, placementPosition.z);
354  float placementDist = Vector2.Distance(cameraXZ, stationXZ);
355  if (placementDist < minPlacementDistance) {
356  float distanceLeft = (minPlacementDistance - placementDist) + workstationObjRadius;
357  Vector2 xzDisplacement = (stationXZ - cameraXZ).normalized * distanceLeft;
358  placementPosition += new Vector3(xzDisplacement[0], 0F, xzDisplacement[1]);
359  }
360 
361  return placementPosition;
362  }
