848  public virtual float GetHoverDistance(Vector3 worldPosition) {
849  float closestComparativeColliderDistance = float.PositiveInfinity;
850  bool hasColliders = false;
851  float testDistance = float.PositiveInfinity;
852 
853  if (rigidbody == null) {
854  // The Interaction Object is probably being destroyed, or is otherwise in an
855  // invalid state.
856  return float.PositiveInfinity;
857  }
858 
859  foreach (var collider in _interactionColliders) {
860  if (!hasColliders) hasColliders = true;
861 
862  if (collider is MeshCollider) {
863  // Native, faster ClosestPoint, but no support for off-center colliders; use to
864  // support MeshColliders.
865  testDistance = (Physics.ClosestPoint(worldPosition,
866  collider,
867  collider.attachedRigidbody.position,
868  collider.attachedRigidbody.rotation)
869  - worldPosition).magnitude;
870  }
871  // Custom, slower ClosestPoint
872  else {
873  // Note: Should be using rigidbody position instead of transform; this will
874  // cause problems when colliders are moving very fast (one-frame delay).
875  testDistance = (collider.transform.TransformPoint(
876  collider.ClosestPointOnSurface(
877  collider.transform.InverseTransformPoint(worldPosition)))
878  - worldPosition).magnitude;
879  }
880 
881  if (testDistance < closestComparativeColliderDistance) {
882  closestComparativeColliderDistance = testDistance;
883  }
884  }
885 
886  if (!hasColliders) {
887  return (this.rigidbody.position - worldPosition).magnitude;
888  }
889  else {
890  return closestComparativeColliderDistance;
891  }
892  }
