495  private void refreshHoverState(HashSet<IInteractionBehaviour> hoverCandidates) {
496  // Prepare data from last frame for hysteresis later on.
497  int primaryHoverPointIdxLastFrame = _primaryHoveredLastFrame != null ? _primaryHoverPointIdx : -1;
498 
499  _hoveredObjects.Clear();
500 
501  IInteractionBehaviour lockedPrimaryHoveredObject = null;
502  if (primaryHoverLocked) {
503  lockedPrimaryHoveredObject = _primaryHoveredObject;
504  }
505  _primaryHoveredObject = null;
506  _primaryHoverDistance = float.PositiveInfinity;
507  _primaryHoverPointIdx = -1;
508  _perPointPrimaryHovered.Clear();
509  _perPointPrimaryHoverDistance.Clear();
510  for (int i = 0; i < primaryHoverPoints.Count; i++) {
511  _perPointPrimaryHovered.Add(null);
512  _perPointPrimaryHoverDistance.Add(float.PositiveInfinity);
513  }
514 
515  // We can only update hover information if there's tracked data.
516  if (!isTracked) return;
517 
518  // Determine values to apply hysteresis to the primary hover state.
519  float maxNewPrimaryHoverDistance = float.PositiveInfinity;
520  if (_primaryHoveredLastFrame != null && primaryHoverPointIdxLastFrame != -1
521  && primaryHoverPoints[primaryHoverPointIdxLastFrame] != null) {
522 
523  if (_contactBehaviours.ContainsKey(_primaryHoveredLastFrame)) {
524  // If we're actually touching the last primary hover, prevent the primary hover from changing at all.
525  maxNewPrimaryHoverDistance = 0F;
526  }
527  else {
528  float distanceToLastPrimaryHover = _primaryHoveredLastFrame.GetHoverDistance(
529  primaryHoverPoints[primaryHoverPointIdxLastFrame].position);
530  // Otherwise...
531  // The distance to a new object must be even closer than the current primary hover
532  // distance in order for that object to become the new primary hover.
533  maxNewPrimaryHoverDistance = distanceToLastPrimaryHover
534  * distanceToLastPrimaryHover.Map(0.009F, 0.018F, 0.4F, 0.95F);
535 
536  }
537  }
538 
539  foreach (IInteractionBehaviour behaviour in hoverCandidates) {
540  // All hover candidates automatically count as hovered.
541  _hoveredObjects.Add(behaviour);
542 
543  // Some objects can ignore consideration for primary hover as an
544  // optimization, since it can require a lot of distance checks.
545  if (behaviour.ignorePrimaryHover) continue;
546 
547  // Do further processing to determine the primary hover if primary hover isn't
548  // locked.
549  else if (!primaryHoverLocked) {
550  processPrimaryHover(behaviour, maxNewPrimaryHoverDistance);
551  }
552  }
553 
554  // If the primary hover is locked, we need to process primary hover specifically
555  // for the locked object.
556  if (primaryHoverLocked && lockedPrimaryHoveredObject != null) {
557  processPrimaryHover(lockedPrimaryHoveredObject, float.PositiveInfinity);
558  }
559  }

561  private void processPrimaryHover(IInteractionBehaviour behaviour, float maxNewPrimaryHoverDistance) {
562  // Check against all positions currently registered as primary hover points,
563  // finding the closest one and updating hover data accordingly.
564  float shortestPointDistance = float.PositiveInfinity;
565  for (int i = 0; i < primaryHoverPoints.Count; i++) {
566  var primaryHoverPoint = primaryHoverPoints[i];
567 
568  // It's possible to disable primary hover points to ignore them for hover
569  // consideration.
570  if (primaryHoverPoint == null) continue;
571  if (!primaryHoverPoint.gameObject.activeInHierarchy) continue;
572 
573  // Skip non-index fingers for InteractionHands if they aren't extended.
574  if (intHand != null) {
575  if (!(intHand.leapHand).Fingers[i].IsExtended && i != 1) { continue; }
576  }
577 
578  // Check primary hover for the primary hover point.
579  float behaviourDistance = GetHoverDistance(primaryHoverPoint.position, behaviour);
580  if (behaviourDistance < shortestPointDistance) {
581 
582  // This is the closest behaviour to this primary hover point.
583  _perPointPrimaryHovered[i] = behaviour;
584  _perPointPrimaryHoverDistance[i] = behaviourDistance;
585  shortestPointDistance = behaviourDistance;
586 
587  if (primaryHoverLocked) {
588  // If primary hover is locked, there's only one object to consider,
589  // and the current primary hover point is the closest one to the object
590  // so far, so update primary hover accordingly.
591  _primaryHoveredObject = _perPointPrimaryHovered[i]; // (redundant)
592  _primaryHoverDistance = _perPointPrimaryHoverDistance[i];
593  _primaryHoverPointIdx = i;
594  }
595  else if (shortestPointDistance < _primaryHoverDistance
596  && (behaviour == _primaryHoveredLastFrame || behaviourDistance < maxNewPrimaryHoverDistance)) {
597 
598  // This is the closest behaviour to ANY primary hover point, and the
599  // distance is less than the hysteresis distance to transition away from
600  // the previous primary hovered object.
601  _primaryHoveredObject = _perPointPrimaryHovered[i];
602  _primaryHoverDistance = _perPointPrimaryHoverDistance[i];
603  _primaryHoverPointIdx = i;
604  }
605  }
606  }
607  }

1036  private void updateContactBone(int contactBoneIndex, Vector3 targetPosition, Quaternion targetRotation) {
1037  ContactBone contactBone = contactBones[contactBoneIndex];
1038  Rigidbody body = contactBone.rigidbody;
1039 
1040  // Infer ahead if the Interaction Manager has a moving frame of reference.
1041  //manager.TransformAheadByFixedUpdate(targetPosition, targetRotation, out targetPosition, out targetRotation);
1042 
1043  // Set a fixed rotation for bones; otherwise most friction is lost
1044  // as any capsule or spherical bones will roll on contact.
1045  using (new ProfilerSample("updateContactBone: MoveRotation")) {
1046  body.MoveRotation(targetRotation);
1047  }
1048 
1049  // Calculate how far off its target the contact bone is.
1050  float errorDistance = 0f;
1051  float errorFraction = 0f;
1052  using (new ProfilerSample("updateContactBone: errorDistance, errorFraction")) {
1053  Vector3 lastTargetPositionTransformedAhead = contactBone.lastTargetPosition;
1054  if (manager.hasMovingFrameOfReference) {
1055  manager.TransformAheadByFixedUpdate(contactBone.lastTargetPosition, out lastTargetPositionTransformedAhead);
1056  }
1057  errorDistance = Vector3.Distance(lastTargetPositionTransformedAhead, body.position);
1058  errorFraction = errorDistance / contactBone.width;
1059  }
1060 
1061  // Adjust the mass of the contact bone based on the mass of
1062  // the object it is currently touching.
1063  float speed = 0f;
1064  using (new ProfilerSample("updateContactBone: speed & massScale")) {
1065  speed = velocity.magnitude;
1066  float massScale = Mathf.Clamp(1.0F - (errorFraction * 2.0F), 0.1F, 1.0F)
1067  * Mathf.Clamp(speed * 10F, 1F, 10F);
1068  body.mass = massScale * contactBone._lastObjectTouchedAdjustedMass;
1069  }
1070 
1071  // Potentially enable Soft Contact if our error is too large.
1072  using (new ProfilerSample("updateContactBone: maybe enable Soft Contact")) {
1073  if (!_softContactEnabled && errorDistance >= softContactDislocationDistance
1074  && speed < 1.5F
1075  /* && boneArrayIndex != NUM_FINGERS * BONES_PER_FINGER */) {
1076  EnableSoftContact();
1077  }
1078  }
1079 
1080  // Attempt to move the contact bone to its target position and rotation
1081  // by setting its target velocity and angular velocity. Include a "deadzone"
1082  // for position to avoid tiny vibrations.
1083  using (new ProfilerSample("updateContactBone: Move to target, with deadzone")) {
1084  float deadzone = Mathf.Min(DEAD_ZONE_FRACTION * contactBone.width, 0.01F * scale);
1085  Vector3 delta = (targetPosition - body.position);
1086  float deltaMag = delta.magnitude;
1087  if (deltaMag <= deadzone) {
1088  body.velocity = Vector3.zero;
1089  contactBone.lastTargetPosition = body.position;
1090  }
1091  else {
1092  delta *= (deltaMag - deadzone) / deltaMag;
1093  contactBone.lastTargetPosition = body.position + delta;
1094 
1095  Vector3 targetVelocity = delta / Time.fixedDeltaTime;
1096  float targetVelocityMag = targetVelocity.magnitude;
1097  body.velocity = (targetVelocity / targetVelocityMag)
1098  * Mathf.Clamp(targetVelocityMag, 0F, 100F);
1099  }
1100  Quaternion deltaRot = targetRotation * Quaternion.Inverse(body.rotation);
1101  body.angularVelocity = PhysicsUtility.ToAngularVelocity(deltaRot, Time.fixedDeltaTime);
1102  }
