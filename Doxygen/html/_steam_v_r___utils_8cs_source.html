20  public static Quaternion Slerp(Quaternion A, Quaternion B, float time)
21  {
22  float cosom = Mathf.Clamp(A.x * B.x + A.y * B.y + A.z * B.z + A.w * B.w, -1.0f, 1.0f);
23  if (cosom < 0.0f)
24  {
25  B = new Quaternion(-B.x, -B.y, -B.z, -B.w);
26  cosom = -cosom;
27  }
28 
29  float sclp, sclq;
30  if ((1.0f - cosom) > 0.0001f)
31  {
32  float omega = Mathf.Acos(cosom);
33  float sinom = Mathf.Sin(omega);
34  sclp = Mathf.Sin((1.0f - time) * omega) / sinom;
35  sclq = Mathf.Sin(time * omega) / sinom;
36  }
37  else
38  {
39  // "from" and "to" very close, so do linear interp
40  sclp = 1.0f - time;
41  sclq = time;
42  }
43 
44  return new Quaternion(
45  sclp * A.x + sclq * B.x,
46  sclp * A.y + sclq * B.y,
47  sclp * A.z + sclq * B.z,
48  sclp * A.w + sclq * B.w);
49  }

400  public static void TakeStereoScreenshot(uint screenshotHandle, GameObject target, int cellSize, float ipd, ref string previewFilename, ref string VRFilename)
401  {
402  const int width = 4096;
403  const int height = width / 2;
404  const int halfHeight = height / 2;
405 
406  Texture2D texture = new Texture2D(width, height * 2, TextureFormat.ARGB32, false);
407 
408  System.Diagnostics.Stopwatch timer = new System.Diagnostics.Stopwatch();
409 
410  Camera tempCamera = null;
411 
412  timer.Start();
413 
414  Camera camera = target.GetComponent<Camera>();
415  if (camera == null)
416  {
417  if (tempCamera == null)
418  tempCamera = new GameObject().AddComponent<Camera>();
419  camera = tempCamera;
420  }
421 
422  // Render preview texture
423  const int previewWidth = 2048;
424  const int previewHeight = 2048;
425  Texture2D previewTexture = new Texture2D(previewWidth, previewHeight, TextureFormat.ARGB32, false);
426  RenderTexture targetPreviewTexture = new RenderTexture(previewWidth, previewHeight, 24);
427 
428  RenderTexture oldTargetTexture = camera.targetTexture;
429  bool oldOrthographic = camera.orthographic;
430  float oldFieldOfView = camera.fieldOfView;
431  float oldAspect = camera.aspect;
432  StereoTargetEyeMask oldstereoTargetEye = camera.stereoTargetEye;
433  camera.stereoTargetEye = StereoTargetEyeMask.None;
434  camera.fieldOfView = 60.0f;
435  camera.orthographic = false;
436  camera.targetTexture = targetPreviewTexture;
437  camera.aspect = 1.0f;
438  camera.Render();
439 
440  // copy preview texture
441  RenderTexture.active = targetPreviewTexture;
442  previewTexture.ReadPixels(new Rect(0, 0, targetPreviewTexture.width, targetPreviewTexture.height), 0, 0);
443  RenderTexture.active = null;
444  camera.targetTexture = null;
445  Object.DestroyImmediate(targetPreviewTexture);
446 
447  SteamVR_SphericalProjection fx = camera.gameObject.AddComponent<SteamVR_SphericalProjection>();
448 
449  Vector3 oldPosition = target.transform.localPosition;
450  Quaternion oldRotation = target.transform.localRotation;
451  Vector3 basePosition = target.transform.position;
452  Quaternion baseRotation = Quaternion.Euler(0, target.transform.rotation.eulerAngles.y, 0);
453 
454  Transform transform = camera.transform;
455 
456  int vTotal = halfHeight / cellSize;
457  float dv = 90.0f / vTotal; // vertical degrees per segment
458  float dvHalf = dv / 2.0f;
459 
460  RenderTexture targetTexture = new RenderTexture(cellSize, cellSize, 24);
461  targetTexture.wrapMode = TextureWrapMode.Clamp;
462  targetTexture.antiAliasing = 8;
463 
464  camera.fieldOfView = dv;
465  camera.orthographic = false;
466  camera.targetTexture = targetTexture;
467  camera.aspect = oldAspect;
468  camera.stereoTargetEye = StereoTargetEyeMask.None;
469 
470  // Render sections of a sphere using a rectilinear projection
471  // and resample using a sphereical projection into a single panorama
472  // texture per eye. We break into sections in order to keep the eye
473  // separation similar around the sphere. Rendering alternates between
474  // top and bottom sections, sweeping horizontally around the sphere,
475  // alternating left and right eyes.
476  for (int v = 0; v < vTotal; v++)
477  {
478  float pitch = 90.0f - (v * dv) - dvHalf;
479  int uTotal = width / targetTexture.width;
480  float du = 360.0f / uTotal; // horizontal degrees per segment
481  float duHalf = du / 2.0f;
482 
483  int vTarget = v * halfHeight / vTotal;
484 
485  for (int i = 0; i < 2; i++) // top, bottom
486  {
487  if (i == 1)
488  {
489  pitch = -pitch;
490  vTarget = height - vTarget - cellSize;
491  }
492 
493  for (int u = 0; u < uTotal; u++)
494  {
495  float yaw = -180.0f + (u * du) + duHalf;
496 
497  int uTarget = u * width / uTotal;
498 
499  int vTargetOffset = 0;
500  float xOffset = -ipd / 2 * Mathf.Cos(pitch * Mathf.Deg2Rad);
501 
502  for (int j = 0; j < 2; j++) // left, right
503  {
504  if (j == 1)
505  {
506  vTargetOffset = height;
507  xOffset = -xOffset;
508  }
509 
510  Vector3 offset = baseRotation * Quaternion.Euler(0, yaw, 0) * new Vector3(xOffset, 0, 0);
511  transform.position = basePosition + offset;
512 
513  Quaternion direction = Quaternion.Euler(pitch, yaw, 0.0f);
514  transform.rotation = baseRotation * direction;
515 
516  // vector pointing to center of this section
517  Vector3 N = direction * Vector3.forward;
518 
519  // horizontal span of this section in degrees
520  float phi0 = yaw - (du / 2);
521  float phi1 = phi0 + du;
522 
523  // vertical span of this section in degrees
524  float theta0 = pitch + (dv / 2);
525  float theta1 = theta0 - dv;
526 
527  float midPhi = (phi0 + phi1) / 2;
528  float baseTheta = Mathf.Abs(theta0) < Mathf.Abs(theta1) ? theta0 : theta1;
529 
530  // vectors pointing to corners of image closes to the equator
531  Vector3 V00 = Quaternion.Euler(baseTheta, phi0, 0.0f) * Vector3.forward;
532  Vector3 V01 = Quaternion.Euler(baseTheta, phi1, 0.0f) * Vector3.forward;
533 
534  // vectors pointing to top and bottom midsection of image
535  Vector3 V0M = Quaternion.Euler(theta0, midPhi, 0.0f) * Vector3.forward;
536  Vector3 V1M = Quaternion.Euler(theta1, midPhi, 0.0f) * Vector3.forward;
537 
538  // intersection points for each of the above
539  Vector3 P00 = V00 / Vector3.Dot(V00, N);
540  Vector3 P01 = V01 / Vector3.Dot(V01, N);
541  Vector3 P0M = V0M / Vector3.Dot(V0M, N);
542  Vector3 P1M = V1M / Vector3.Dot(V1M, N);
543 
544  // calculate basis vectors for plane
545  Vector3 P00_P01 = P01 - P00;
546  Vector3 P0M_P1M = P1M - P0M;
547 
548  float uMag = P00_P01.magnitude;
549  float vMag = P0M_P1M.magnitude;
550 
551  float uScale = 1.0f / uMag;
552  float vScale = 1.0f / vMag;
553 
554  Vector3 uAxis = P00_P01 * uScale;
555  Vector3 vAxis = P0M_P1M * vScale;
556 
557  // update material constant buffer
558  fx.Set(N, phi0, phi1, theta0, theta1,
559  uAxis, P00, uScale,
560  vAxis, P0M, vScale);
561 
562  camera.aspect = uMag / vMag;
563  camera.Render();
564 
565  RenderTexture.active = targetTexture;
566  texture.ReadPixels(new Rect(0, 0, targetTexture.width, targetTexture.height), uTarget, vTarget + vTargetOffset);
567  RenderTexture.active = null;
568  }
569 
570  // Update progress
571  float progress = (float)(v * (uTotal * 2.0f) + u + i * uTotal) / (float)(vTotal * (uTotal * 2.0f));
572  OpenVR.Screenshots.UpdateScreenshotProgress(screenshotHandle, progress);
573  }
574  }
575  }
576 
577  // 100% flush
578  OpenVR.Screenshots.UpdateScreenshotProgress(screenshotHandle, 1.0f);
579 
580  // Save textures to disk.
581  // Add extensions
582  previewFilename += ".png";
583  VRFilename += ".png";
584 
585  // Preview
586  previewTexture.Apply();
587  System.IO.File.WriteAllBytes(previewFilename, previewTexture.EncodeToPNG());
588 
589  // VR
590  texture.Apply();
591  System.IO.File.WriteAllBytes(VRFilename, texture.EncodeToPNG());
592 
593  // Cleanup.
594  if (camera != tempCamera)
595  {
596  camera.targetTexture = oldTargetTexture;
597  camera.orthographic = oldOrthographic;
598  camera.fieldOfView = oldFieldOfView;
599  camera.aspect = oldAspect;
600  camera.stereoTargetEye = oldstereoTargetEye;
601 
602  target.transform.localPosition = oldPosition;
603  target.transform.localRotation = oldRotation;
604  }
605  else
606  {
607  tempCamera.targetTexture = null;
608  }
609 
610  Object.DestroyImmediate(targetTexture);
611  Object.DestroyImmediate(fx);
612 
613  timer.Stop();
614  Debug.Log(string.Format("<b>[SteamVR]</b> Screenshot took {0} seconds.", timer.Elapsed));
615 
616  if (tempCamera != null)
617  {
618  Object.DestroyImmediate(tempCamera.gameObject);
619  }
620 
621  Object.DestroyImmediate(previewTexture);
622  Object.DestroyImmediate(texture);
623  }
