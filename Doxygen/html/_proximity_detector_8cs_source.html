134  IEnumerator proximityWatcher(){
135  bool proximityState = false;
136  float onSquared, offSquared; //Use squared distances to avoid taking square roots
137  while(true){
138  onSquared = OnDistance * OnDistance;
139  offSquared = OffDistance * OffDistance;
140  if(_currentObj != null){
141  if(distanceSquared(_currentObj) > offSquared){
142  _currentObj = null;
143  proximityState = false;
144  }
145  } else {
146  if (UseLayersNotList) {
147  Collider[] nearby = Physics.OverlapSphere(transform.position, OnDistance, Layer);
148  if(nearby.Length > 0) {
149  _currentObj = nearby[0].gameObject;
150  proximityState = true;
151  OnProximity.Invoke(_currentObj);
152  }
153  } else {
154  for (int obj = 0; obj < TargetObjects.Length; obj++) {
155  GameObject target = TargetObjects[obj];
156  if (distanceSquared(target) < onSquared) {
157  _currentObj = target;
158  proximityState = true;
159  OnProximity.Invoke(_currentObj);
160  break; // pick first match
161  }
162  }
163  }
164  }
165  if(proximityState){
166  Activate();
167  } else {
168  Deactivate();
169  }
170  yield return new WaitForSeconds(Period);
171  }
172  }
