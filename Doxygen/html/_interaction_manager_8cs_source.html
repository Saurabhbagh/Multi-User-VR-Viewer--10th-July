346  private HashSet<InteractionController> _activeControllersBuffer = new HashSet<InteractionController>();
347  private HashSet<InteractionController> _hoverControllersBuffer = new HashSet<InteractionController>();

349  private HashSet<InteractionController> _graspingControllersBuffer = new HashSet<InteractionController>();
350 
351  private void fixedUpdateInteractionControllers() {
352 
353  _hoverControllersBuffer.Clear();
354  _contactControllersBuffer.Clear();
355  _graspingControllersBuffer.Clear();
356  _activeControllersBuffer.Clear();
357  foreach (var controller in interactionControllers) {
358  if (!controller.isActiveAndEnabled) continue;
359 
360  _activeControllersBuffer.Add(controller);
361 
362  if (controller.hoverEnabled) _hoverControllersBuffer.Add(controller);
363  if (controller.contactEnabled) _contactControllersBuffer.Add(controller);
364  if (controller.graspingEnabled) _graspingControllersBuffer.Add(controller);
365  }
366 
367  using (new ProfilerSample("Fixed Update Controllers (General Update)")) {
368  // Perform general controller update, for controller collider and point
369  // representations.
370  foreach (var controller in _activeControllersBuffer) {
371  if (!controller.isActiveAndEnabled) continue;
372  (controller as IInternalInteractionController).FixedUpdateController();
373  }
374  }
375 
376  using (new ProfilerSample("Fixed Update Controllers (Interaction State and Callbacks)")) {
377 
378  /*
379  * Interactions are checked here in a very specific manner so that interaction
380  * callbacks always occur in a strict order and interaction object state is
381  * always updated directly before the relevant callbacks occur.
382  *
383  * Interaction callbacks will only occur outside this order if a script
384  * manually forces interaction state-changes; for example, calling
385  * interactionController.ReleaseGrasp() will immediately call
386  * interactionObject.OnPerControllerGraspEnd() on the formerly grasped object.
387  *
388  * Callback order:
389  * - Suspension (when a grasped object's grasping controller loses tracking)
390  * - Just-Ended Interactions (Grasps, then Contacts, then Hovers)
391  * - Just-Begun Interactions (Hovers, then Contacts, then Grasps)
392  * - Sustained Interactions (Hovers, then Contacts, then Grasps)
393  */
394 
395  // Suspension //
396 
397  // Check controllers beginning object suspension.
398  foreach (var controller in _graspingControllersBuffer) {
399  IInteractionBehaviour suspendedObj;
400  if ((controller as IInternalInteractionController).CheckSuspensionBegin(out suspendedObj)) {
401  suspendedObj.BeginSuspension(controller);
402  }
403  }
404 
405  // Check controllers ending object suspension.
406  foreach (var controller in _graspingControllersBuffer) {
407  IInteractionBehaviour resumedObj;
408  if ((controller as IInternalInteractionController).CheckSuspensionEnd(out resumedObj)) {
409  resumedObj.EndSuspension(controller);
410  }
411  }
412 
413  // Ending Interactions //
414 
415  checkEndingGrasps(_graspingControllersBuffer);
416  checkEndingContacts(_contactControllersBuffer);
417  checkEndingPrimaryHovers(_hoverControllersBuffer);
418  checkEndingHovers(_hoverControllersBuffer);
419 
420  // Beginning Interactions //
421 
422  checkBeginningHovers(_hoverControllersBuffer);
423  checkBeginningPrimaryHovers(_hoverControllersBuffer);
424  checkBeginningContacts(_contactControllersBuffer);
425  checkBeginningGrasps(_graspingControllersBuffer);
426 
427  // Sustained Interactions //
428 
429  checkSustainingHovers(_hoverControllersBuffer);
430  checkSustainingPrimaryHovers(_hoverControllersBuffer);
431  checkSustainingContacts(_contactControllersBuffer);
432  checkSustainingGrasps(_graspingControllersBuffer);
433 
434  }
435  }

574  [ThreadStatic]
575  private static Dictionary<IInteractionBehaviour, List<InteractionController>> s_objControllersMap = new Dictionary<IInteractionBehaviour, List<InteractionController>>();
576 
580  private void remapInteractionObjectStateChecks(ReadonlyHashSet<InteractionController> controllers,
581  StateChangeCheckFunc stateCheckFunc,
582  Action<IInteractionBehaviour, List<InteractionController>> actionPerInteractionObject) {
583 
584  // Ensure the object->controllers buffer is non-null (ThreadStatic quirk) and clean.
585  if (s_objControllersMap == null) s_objControllersMap = new Dictionary<IInteractionBehaviour, List<InteractionController>>();
586  s_objControllersMap.Clear();
587 
588  // In a nutshell, this remaps methods per-controller that output an interaction object if the controller changed that object's state
589  // to methods per-object with all of the controllers for which the check produced a state-change.
590  foreach (var controller in controllers) {
591  IInteractionBehaviour objectWhoseStateChanged;
592  if (stateCheckFunc(controller, out objectWhoseStateChanged)) {
593  if (!s_objControllersMap.ContainsKey(objectWhoseStateChanged)) {
594  s_objControllersMap[objectWhoseStateChanged] = Pool<List<InteractionController>>.Spawn();
595  }
596  s_objControllersMap[objectWhoseStateChanged].Add(controller);
597  }
598  }
599  // Finally, iterate through each (object, controllers) pair and call the action for each pair
600  foreach (var objControllesPair in s_objControllersMap) {
601  actionPerInteractionObject(objControllesPair.Key, objControllesPair.Value);
602 
603  // Clear each controllers list and return it to the list pool.
604  objControllesPair.Value.Clear();
605  Pool<List<InteractionController>>.Recycle(objControllesPair.Value);
606  }
607  }
608 
612  private void remapMultiInteractionObjectStateChecks(ReadonlyHashSet<InteractionController> controllers,
613  MultiStateChangeCheckFunc multiObjectStateCheckFunc,
614  Action<IInteractionBehaviour, List<InteractionController>> actionPerInteractionObject) {
615  // Ensure object<->controllers buffer is non-null (ThreadStatic quirk) and clean.
616  if (s_objControllersMap == null) s_objControllersMap = new Dictionary<IInteractionBehaviour, List<InteractionController>>();
617  s_objControllersMap.Clear();
618 
619  // In a nutshell, this remaps methods per-controller that output multiple interaction objects if the controller changed those objects' states
620  // to methods per-object with all of the controllers for which the check produced a state-change.
621  foreach (var controller in controllers) {
622  HashSet<IInteractionBehaviour> stateChangedObjects;
623  if (multiObjectStateCheckFunc(controller, out stateChangedObjects)) {
624  foreach (var stateChangedObject in stateChangedObjects) {
625  if (!s_objControllersMap.ContainsKey(stateChangedObject)) {
626  s_objControllersMap[stateChangedObject] = Pool<List<InteractionController>>.Spawn();
627  }
628  s_objControllersMap[stateChangedObject].Add(controller);
629  }
630  }
631  }
632  // Finally, iterate through each (object, controllers) pair and call the action for each pair
633  foreach (var objControllersPair in s_objControllersMap) {
634  actionPerInteractionObject(objControllersPair.Key, objControllersPair.Value);
635 
636  // Clear each controllers list and return it to the list pool.
637  objControllersPair.Value.Clear();
638  Pool<List<InteractionController>>.Recycle(objControllersPair.Value);
639  }
640  }
