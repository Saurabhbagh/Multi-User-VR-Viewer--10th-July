148  private int GetSphereColliderResults(Vector3 position, ref Collider[] resultsBuffer) {
149  using (new ProfilerSample("GetSphereColliderResults()")) {
150 
151  int layerMask = activationLayerFunction == null ? activationLayerMask : activationLayerFunction();
152  int overlapCount = 0;
153  while (true) {
154  overlapCount = Physics.OverlapSphereNonAlloc(position,
155  activationRadius,
156  resultsBuffer,
157  layerMask,
158  QueryTriggerInteraction.Collide);
159  if (overlapCount < resultsBuffer.Length) {
160  break;
161  } else {
162  // Non-allocating sphere-overlap fills the existing _resultsBuffer array.
163  // If the output overlapCount is equal to the array's length, there might be more collision results
164  // that couldn't be returned because the array wasn't large enough, so try again with increased length.
165  // The _in, _out argument setup allows allocating a new array from within this function.
166  resultsBuffer = new Collider[resultsBuffer.Length * 2];
167  }
168  }
169  return overlapCount;
170  }
