148  private void refreshAttachmentHands() {
149  // If we're a prefab, we'll be unable to set parent transforms, so we shouldn't create new objects in general.
150  bool isPrefab = false;
151  #if UNITY_EDITOR
152  isPrefab = getIsPrefab();
153  #endif
154 
155  // If necessary, generate a left and right AttachmentHand.
156  if (_attachmentHands == null || _attachmentHands.Length == 0 || (_attachmentHands[0] == null || _attachmentHands[1] == null)) {
157  _attachmentHands = new AttachmentHand[2];
158 
159  // Try to use existing AttachmentHand objects first.
160  foreach (Transform child in this.transform.GetChildren()) {
161  var attachmentHand = child.GetComponent<AttachmentHand>();
162  if (attachmentHand != null) {
163  _attachmentHands[attachmentHand.chirality == Chirality.Left ? 0 : 1] = attachmentHand;
164  }
165  }
166 
167  // If we are a prefab and there are missing AttachmentHands, we have to return early.
168  // We can't set parent transforms while a prefab. We're only OK if we already have attachmentHand
169  // objects and their parents are properly set.
170  if (isPrefab && (_attachmentHands[0] == null || _attachmentHands[0].transform.parent != this.transform
171  || _attachmentHands[1] == null || _attachmentHands[1].transform.parent != this.transform)) {
172  return;
173  }
174 
175  // Construct any missing AttachmentHand objects.
176  if (_attachmentHands[0] == null) {
177  GameObject obj = new GameObject();
178  #if UNITY_EDITOR
179  Undo.RegisterCreatedObjectUndo(obj, "Created GameObject");
180  #endif
181  _attachmentHands[0] = obj.AddComponent<AttachmentHand>();
182  _attachmentHands[0].chirality = Chirality.Left;
183  }
184  _attachmentHands[0].gameObject.name = "Attachment Hand (Left)";
185  if (_attachmentHands[0].transform.parent != this.transform) _attachmentHands[0].transform.parent = this.transform;
186 
187  if (_attachmentHands[1] == null) {
188  GameObject obj = new GameObject();
189  #if UNITY_EDITOR
190  Undo.RegisterCreatedObjectUndo(obj, "Created GameObject");
191  #endif
192  _attachmentHands[1] = obj.AddComponent<AttachmentHand>();
193  _attachmentHands[1].chirality = Chirality.Right;
194  }
195  _attachmentHands[1].gameObject.name = "Attachment Hand (Right)";
196  if (_attachmentHands[1].transform.parent != this.transform) _attachmentHands[1].transform.parent = this.transform;

198  // Organize left hand first in sibling order.
