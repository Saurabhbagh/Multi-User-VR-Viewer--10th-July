310  private void UpdatePointer()
311  {
312  Vector3 pointerStart = pointerStartTransform.position;
313  Vector3 pointerEnd;
314  Vector3 pointerDir = pointerStartTransform.forward;
315  bool hitSomething = false;
316  bool showPlayAreaPreview = false;
317  Vector3 playerFeetOffset = player.trackingOriginTransform.position - player.feetPositionGuess;
318 
319  Vector3 arcVelocity = pointerDir * arcDistance;
320 
321  TeleportMarkerBase hitTeleportMarker = null;
322 
323  //Check pointer angle
324  float dotUp = Vector3.Dot( pointerDir, Vector3.up );
325  float dotForward = Vector3.Dot( pointerDir, player.hmdTransform.forward );
326  bool pointerAtBadAngle = false;
327  if ( ( dotForward > 0 && dotUp > 0.75f ) || ( dotForward < 0.0f && dotUp > 0.5f ) )
328  {
329  pointerAtBadAngle = true;
330  }
331 
332  //Trace to see if the pointer hit anything
333  RaycastHit hitInfo;
334  teleportArc.SetArcData( pointerStart, arcVelocity, true, pointerAtBadAngle );
335  if ( teleportArc.DrawArc( out hitInfo ) )
336  {
337  hitSomething = true;
338  hitTeleportMarker = hitInfo.collider.GetComponentInParent<TeleportMarkerBase>();
339  }
340 
341  if ( pointerAtBadAngle )
342  {
343  hitTeleportMarker = null;
344  }
345 
346  HighlightSelected( hitTeleportMarker );
347 
348  if ( hitTeleportMarker != null ) //Hit a teleport marker
349  {
350  if ( hitTeleportMarker.locked )
351  {
352  teleportArc.SetColor( pointerLockedColor );
353 #if (UNITY_5_4)
354  pointerLineRenderer.SetColors( pointerLockedColor, pointerLockedColor );
355 #else
356  pointerLineRenderer.startColor = pointerLockedColor;
357  pointerLineRenderer.endColor = pointerLockedColor;
358 #endif
359  destinationReticleTransform.gameObject.SetActive( false );
360  }
361  else
362  {
363  teleportArc.SetColor( pointerValidColor );
364 #if (UNITY_5_4)
365  pointerLineRenderer.SetColors( pointerValidColor, pointerValidColor );
366 #else
367  pointerLineRenderer.startColor = pointerValidColor;
368  pointerLineRenderer.endColor = pointerValidColor;
369 #endif
370  destinationReticleTransform.gameObject.SetActive( hitTeleportMarker.showReticle );
371  }
372 
373  offsetReticleTransform.gameObject.SetActive( true );
374 
375  invalidReticleTransform.gameObject.SetActive( false );
376 
377  pointedAtTeleportMarker = hitTeleportMarker;
378  pointedAtPosition = hitInfo.point;
379 
380  if ( showPlayAreaMarker )
381  {
382  //Show the play area marker if this is a teleport area
383  TeleportArea teleportArea = pointedAtTeleportMarker as TeleportArea;
384  if ( teleportArea != null && !teleportArea.locked && playAreaPreviewTransform != null )
385  {
386  Vector3 offsetToUse = playerFeetOffset;
387 
388  //Adjust the actual offset to prevent the play area marker from moving too much
389  if ( !movedFeetFarEnough )
390  {
391  float distanceFromStartingOffset = Vector3.Distance( playerFeetOffset, startingFeetOffset );
392  if ( distanceFromStartingOffset < 0.1f )
393  {
394  offsetToUse = startingFeetOffset;
395  }
396  else if ( distanceFromStartingOffset < 0.4f )
397  {
398  offsetToUse = Vector3.Lerp( startingFeetOffset, playerFeetOffset, ( distanceFromStartingOffset - 0.1f ) / 0.3f );
399  }
400  else
401  {
402  movedFeetFarEnough = true;
403  }
404  }
405 
406  playAreaPreviewTransform.position = pointedAtPosition + offsetToUse;
407 
408  showPlayAreaPreview = true;
409  }
410  }
411 
412  pointerEnd = hitInfo.point;
413  }
414  else //Hit neither
415  {
416  destinationReticleTransform.gameObject.SetActive( false );
417  offsetReticleTransform.gameObject.SetActive( false );
418 
419  teleportArc.SetColor( pointerInvalidColor );
420 #if (UNITY_5_4)
421  pointerLineRenderer.SetColors( pointerInvalidColor, pointerInvalidColor );
422 #else
423  pointerLineRenderer.startColor = pointerInvalidColor;
424  pointerLineRenderer.endColor = pointerInvalidColor;
425 #endif
426  invalidReticleTransform.gameObject.SetActive( !pointerAtBadAngle );
427 
428  //Orient the invalid reticle to the normal of the trace hit point
429  Vector3 normalToUse = hitInfo.normal;
430  float angle = Vector3.Angle( hitInfo.normal, Vector3.up );
431  if ( angle < 15.0f )
432  {
433  normalToUse = Vector3.up;
434  }
435  invalidReticleTargetRotation = Quaternion.FromToRotation( Vector3.up, normalToUse );
436  invalidReticleTransform.rotation = Quaternion.Slerp( invalidReticleTransform.rotation, invalidReticleTargetRotation, 0.1f );
437 
438  //Scale the invalid reticle based on the distance from the player
439  float distanceFromPlayer = Vector3.Distance( hitInfo.point, player.hmdTransform.position );
440  float invalidReticleCurrentScale = Util.RemapNumberClamped( distanceFromPlayer, invalidReticleMinScaleDistance, invalidReticleMaxScaleDistance, invalidReticleMinScale, invalidReticleMaxScale );
441  invalidReticleScale.x = invalidReticleCurrentScale;
442  invalidReticleScale.y = invalidReticleCurrentScale;
443  invalidReticleScale.z = invalidReticleCurrentScale;
444  invalidReticleTransform.transform.localScale = invalidReticleScale;
445 
446  pointedAtTeleportMarker = null;
447 
448  if ( hitSomething )
449  {
450  pointerEnd = hitInfo.point;
451  }
452  else
453  {
454  pointerEnd = teleportArc.GetArcPositionAtTime( teleportArc.arcDuration );
455  }
456 
457  //Debug floor
458  if ( debugFloor )
459  {
460  floorDebugSphere.gameObject.SetActive( false );
461  floorDebugLine.gameObject.SetActive( false );
462  }
463  }
464 
465  if ( playAreaPreviewTransform != null )
466  {
467  playAreaPreviewTransform.gameObject.SetActive( showPlayAreaPreview );
468  }
469 
470  if ( !showOffsetReticle )
471  {
472  offsetReticleTransform.gameObject.SetActive( false );
473  }
474 
475  destinationReticleTransform.position = pointedAtPosition;
476  invalidReticleTransform.position = pointerEnd;
477  onActivateObjectTransform.position = pointerEnd;
478  onDeactivateObjectTransform.position = pointerEnd;
479  offsetReticleTransform.position = pointerEnd - playerFeetOffset;
480 
481  reticleAudioSource.transform.position = pointedAtPosition;
482 
483  pointerLineRenderer.SetPosition( 0, pointerStart );
484  pointerLineRenderer.SetPosition( 1, pointerEnd );
485  }
