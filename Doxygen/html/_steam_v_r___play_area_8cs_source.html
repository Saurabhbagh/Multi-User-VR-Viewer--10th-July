89  public void BuildMesh()
90  {
91  var rect = new HmdQuad_t();
92  if (!GetBounds(size, ref rect))
93  return;
94 
95  var corners = new HmdVector3_t[] { rect.vCorners0, rect.vCorners1, rect.vCorners2, rect.vCorners3 };
96 
97  vertices = new Vector3[corners.Length * 2];
98  for (int i = 0; i < corners.Length; i++)
99  {
100  var c = corners[i];
101  vertices[i] = new Vector3(c.v0, 0.01f, c.v2);
102  }
103 
104  if (borderThickness == 0.0f)
105  {
106  GetComponent<MeshFilter>().mesh = null;
107  return;
108  }
109 
110  for (int i = 0; i < corners.Length; i++)
111  {
112  int next = (i + 1) % corners.Length;
113  int prev = (i + corners.Length - 1) % corners.Length;
114 
115  var nextSegment = (vertices[next] - vertices[i]).normalized;
116  var prevSegment = (vertices[prev] - vertices[i]).normalized;
117 
118  var vert = vertices[i];
119  vert += Vector3.Cross(nextSegment, Vector3.up) * borderThickness;
120  vert += Vector3.Cross(prevSegment, Vector3.down) * borderThickness;
121 
122  vertices[corners.Length + i] = vert;
123  }

226  public void DrawWireframe()
