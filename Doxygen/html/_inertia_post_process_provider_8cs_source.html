46  public override void ProcessFrame(ref Frame inputFrame) {
47  var leftHand = inputFrame.Hands.Query().FirstOrDefault(h => h.IsLeft);
48  var rightHand = inputFrame.Hands.Query().FirstOrDefault(h => !h.IsLeft);
49 
50  // Frames can potentially come from two time-interwoven sources: Update frames
51  // and FixedUpdate frames. Time is not monotonically increasing frame-to-frame
52  // because FixedUpdates and Updates interweave and occasionally FixedUpdate plays
53  // catch-up, and we interpolate hand data accordingly further up the hand pipeline,
54  // which affects the hand.TimeVisible property we use to simulate our effect
55  // statefully over time.
56  //
57  // To support both Update-time hand data and FixedUpdate-time hand data with a
58  // single stateful post-process, we maintain two independent states for each stream,
59  // which, independently, _are_ going to be monotonically forward-moving in time.
60  if (Time.inFixedTimeStep) {
61  // FixedUpdate hand data.
62  processHand(leftHand,
63  ref _fixedLeftPose, ref _fixedPreviousLeftPose, ref _fixedLeftAge);
64  processHand(rightHand,
65  ref _fixedRightPose, ref _fixedPreviousRightPose, ref _fixedRightAge);
66  }
67  else {

69  processHand(leftHand, ref _leftPose, ref _previousLeftPose, ref _leftAge);

75  private void processHand(Hand hand,
76  ref Pose? maybeCurPose,
77  ref Pose? maybePrevPose,
78  ref float handAge) {
79  if (hand == null) {
80  // Clear state.
81  maybeCurPose = null;
82  maybePrevPose = null;
83  handAge = 0f;
84  }
85  else {
86  var framePose = hand.GetPalmPose();
87 
88  if (!maybeCurPose.HasValue) {
89  // The hand just started being tracked.
90  maybePrevPose = null;
91  maybeCurPose = framePose;
92  }
93  else if (!maybePrevPose.HasValue) {
94  // Have current pose, lack previous pose, just get initial momentum.
95  maybePrevPose = maybeCurPose;
96  maybeCurPose = framePose;
97  }
98  else {
99  // There's enough data to verlet-integrate.
100 
101  // Calculate how much time has passed since we last received hand data.
102  //
103  // As a safety measure, we ensure deltaTime is positive before running our
104  // stateful filter to give the hand momentum. Any post-process could mess with
105  // the TimeVisible property, so we do this to minimize the chance of total
106  // havok.
107  var deltaTime = hand.TimeVisible - handAge;
108  if (deltaTime > 0) {
109  handAge = hand.TimeVisible;
110 
111  var curPose = maybeCurPose.Value;
112  var prevPose = maybePrevPose.Value;
113  integratePose(ref curPose, ref prevPose,
114  targetPose: framePose, deltaTime: deltaTime);
115  hand.SetPalmPose(curPose);
116  maybeCurPose = curPose;
117  maybePrevPose = prevPose;
118  }
119  }
120  }

127  private void integratePose(ref Pose curPose, ref Pose prevPose,
128  Pose targetPose, float deltaTime) {
129  // Calculate motion from prevPose to curPose.
130  var deltaPose = curPose.inverse * prevPose; // prevPose in curPose's local space.
131  deltaPose = new Pose(-deltaPose.position, Quaternion.Inverse(deltaPose.rotation));
132  deltaPose = Pose.Lerp(deltaPose, Pose.identity, damping * deltaTime); // Dampen.
133 
134  // Verlet-integrate curPose based on the delta from prevPose.
135  Pose tempPose = curPose;
136  curPose = curPose * deltaPose;
137  prevPose = tempPose;
138 
139  // Pull the integrated hand toward the target a little bit based on stiffness.
140  curPose = Pose.Lerp(curPose, targetPose, stiffness * deltaTime);
141  }
