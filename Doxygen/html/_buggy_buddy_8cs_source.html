223  public static float FindAngle(Vector3 fromVector, Vector3 toVector, Vector3 upVector)
224  {
225  // If the vector the angle is being calculated to is 0...
226  if (toVector == Vector3.zero)
227  // ... the angle between them is 0.
228  return 0f;
229 
230  // Create a float to store the angle between the facing of the enemy and the direction it's travelling.
231  float angle = Vector3.Angle(fromVector, toVector);
232 
233  // Find the cross product of the two vectors (this will point up if the velocity is to the right of forward).
234  Vector3 normal = Vector3.Cross(fromVector, toVector);
235 
236  // The dot product of the normal with the upVector will be positive if they point in the same direction.
237  angle *= Mathf.Sign(Vector3.Dot(normal, upVector));
238 
239  // We need to convert the angle we've found from degrees to radians.
240  angle *= Mathf.Deg2Rad;
241 
242  return angle;
243  }
