96  private bool detectCycle() {
97  LeapProvider providerA = _inputLeapProvider, providerB = _inputLeapProvider;
98  while (providerA is PostProcessProvider) {
99  providerB = (providerB as PostProcessProvider).inputLeapProvider;
100  if (providerA == providerB) { return true; }
101  else if (!(providerB is PostProcessProvider)) { return false; }
102  providerA = (providerA as PostProcessProvider).inputLeapProvider;
103  providerB = (providerB as PostProcessProvider).inputLeapProvider;
104  if (!(providerB is PostProcessProvider)) { return false; }
105  }
106  return false;
107  }
