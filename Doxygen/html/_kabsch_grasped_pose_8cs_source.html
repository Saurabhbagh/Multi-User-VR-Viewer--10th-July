79  public void GetGraspedPosition(out Vector3 newPosition, out Quaternion newRotation) {
80  _points.Clear(); _refPoints.Clear();
81  Vector3 bodyPosition = _intObj.rigidbody.position;
82  Quaternion bodyRotation = _intObj.rigidbody.rotation;
83  Matrix4x4 it = Matrix4x4.TRS(bodyPosition, bodyRotation, Vector3.one);
84 
85  _controllerCentroid = Vector3.zero; _objectCentroid = Vector3.zero; _manipulatorCount = 0f;
86 
87  foreach (var controllerPointPair in _controllerToPoints) {
88  InteractionController controller = controllerPointPair.Key;
89  PosePointCollection points = _controllerToPoints[controller];
90 
91  for (int i = 0; i < controller.graspManipulatorPoints.Count; i++) {
92 
93  Vector3 originalManipulatorPos = points.GetLocalPosition(i);
94  Vector3 currentManipulatorPos = controller.graspManipulatorPoints[i];
95 
96  // Perform the solve such that the objects' positions are matched to the new
97  // manipulator positions.
98  Vector3 point1 = (it.MultiplyPoint3x4(originalManipulatorPos) - bodyPosition);
99  Vector3 point2 = (currentManipulatorPos - bodyPosition);
100 
101  if (_intObj.isPositionLocked) {
102  // Only rotate the object, pivoting around its origin.
103  _solveMethod = SolveMethod.PivotAroundOrigin;
104  _objectCentroid += point1;
105  _controllerCentroid += point2;
106  _manipulatorCount += 1F;
107  }
108  else {
109  // Do normal Kabsch solve.
110  _solveMethod = SolveMethod.SixDegreeSolve;
111  _points.Add(point1); _refPoints.Add(point2);
112  }
113  }
114  }
115 
116  Matrix4x4 kabschTransform = PerformSolve(bodyPosition);
117 
118  newPosition = bodyPosition + kabschTransform.GetVector3();
119  newRotation = kabschTransform.GetQuaternion() * bodyRotation;
120  }
121 
122  private Matrix4x4 PerformSolve(Vector3 position) {
123  switch (_solveMethod) {
124  case SolveMethod.SixDegreeSolve:
125  return _kabsch.SolveKabsch(_points, _refPoints);
126 
127  case SolveMethod.PivotAroundOrigin:
128  _objectCentroid /= _manipulatorCount;
129  _controllerCentroid /= _manipulatorCount;
130  if (!_objectCentroid.Equals(_controllerCentroid)) {
131  return Matrix4x4.TRS(Vector3.zero, Quaternion.FromToRotation(_objectCentroid, _controllerCentroid), Vector3.one);
132  }
133  else {
134  return Matrix4x4.identity;
135  }
136  default:
137  return _kabsch.SolveKabsch(_points, _refPoints);
138  }
