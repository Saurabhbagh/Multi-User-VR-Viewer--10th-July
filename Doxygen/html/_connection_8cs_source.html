142  public void Start() {
143  if (_isRunning)
144  return;
145 
146  eLeapRS result;
147  if (_leapConnection == IntPtr.Zero) {
148  result = LeapC.CreateConnection(out _leapConnection);
149  if (result != eLeapRS.eLeapRS_Success || _leapConnection == IntPtr.Zero) {
150  reportAbnormalResults("LeapC CreateConnection call was ", result);
151  return;
152  }
153  }
154  result = LeapC.OpenConnection(_leapConnection);
155  if (result != eLeapRS.eLeapRS_Success) {
156  reportAbnormalResults("LeapC OpenConnection call was ", result);
157  return;
158  }
159  // The Allocator must persist the lifetime of the connection
160  if (_pLeapAllocator.allocate == null) {
161  _pLeapAllocator.allocate = MemoryManager.Pin;
162  }
163  if (_pLeapAllocator.deallocate == null) {
164  _pLeapAllocator.deallocate = MemoryManager.Unpin;
165  }
166  LeapC.SetAllocator(_leapConnection, ref _pLeapAllocator);
167 
168  _isRunning = true;
169  AppDomain.CurrentDomain.DomainUnload += (arg1, arg2) => Dispose(true);
170 
171  _polster = new Thread(new ThreadStart(this.processMessages));
172  _polster.Name = "LeapC Worker";
173  _polster.IsBackground = true;
174  _polster.Start();
175  }
176 
177  public void Stop() {
178  if (!_isRunning)
179  return;
180 
181  _isRunning = false;
182 
183  //Very important to close the connection before we try to join the
184  //worker thread! The call to PollConnection can sometimes block,
185  //despite the timeout, causing an attempt to join the thread waiting
186  //forever and preventing the connection from stopping.
187  //
188  //It seems that closing the connection causes PollConnection to
189  //unblock in these cases, so just make sure to close the connection
190  //before trying to join the worker thread.
191  LeapC.CloseConnection(_leapConnection);
192 
193  _polster.Join();
194  }

196  //Run in Polster thread, fills in object queues
197  private void processMessages() {
198  //Only profiling block currently is the Handle Event block
199  const string HANDLE_EVENT_PROFILER_BLOCK = "Handle Event";
200  bool hasBegunProfilingForThread = false;
201 
202  try {
203  eLeapRS result;
204  _leapInit.DispatchOnContext(this, EventContext, new LeapEventArgs(LeapEvent.EVENT_INIT));
205  while (_isRunning) {
206  if (LeapBeginProfilingForThread != null && !hasBegunProfilingForThread) {
207  LeapBeginProfilingForThread(new BeginProfilingForThreadArgs("Worker Thread",
208  HANDLE_EVENT_PROFILER_BLOCK));
209  hasBegunProfilingForThread = true;
210  }
211 
212  LEAP_CONNECTION_MESSAGE _msg = new LEAP_CONNECTION_MESSAGE();
213  uint timeout = 150;
214  result = LeapC.PollConnection(_leapConnection, timeout, ref _msg);
215 
216  if (result != eLeapRS.eLeapRS_Success) {
217  reportAbnormalResults("LeapC PollConnection call was ", result);
218  continue;
219  }
220 
221  if (LeapBeginProfilingBlock != null && hasBegunProfilingForThread) {
222  LeapBeginProfilingBlock(new BeginProfilingBlockArgs(HANDLE_EVENT_PROFILER_BLOCK));
223  }
224 
225  switch (_msg.type) {
226  case eLeapEventType.eLeapEventType_None:
227  break;
228 
229  case eLeapEventType.eLeapEventType_Connection:
230  LEAP_CONNECTION_EVENT connection_evt;
231  StructMarshal<LEAP_CONNECTION_EVENT>.PtrToStruct(_msg.eventStructPtr, out connection_evt);
232  handleConnection(ref connection_evt);
233  break;
234  case eLeapEventType.eLeapEventType_ConnectionLost:
235  LEAP_CONNECTION_LOST_EVENT connection_lost_evt;
236  StructMarshal<LEAP_CONNECTION_LOST_EVENT>.PtrToStruct(_msg.eventStructPtr, out connection_lost_evt);
237  handleConnectionLost(ref connection_lost_evt);
238  break;
239 
240  case eLeapEventType.eLeapEventType_Device:
241  LEAP_DEVICE_EVENT device_evt;
242  StructMarshal<LEAP_DEVICE_EVENT>.PtrToStruct(_msg.eventStructPtr, out device_evt);
243  handleDevice(ref device_evt);
244  break;
245 
246  // Note that unplugging a device generates an eLeapEventType_DeviceLost event
247  // message, not a failure message. DeviceLost is further down.
248  case eLeapEventType.eLeapEventType_DeviceFailure:
249  LEAP_DEVICE_FAILURE_EVENT device_failure_evt;
250  StructMarshal<LEAP_DEVICE_FAILURE_EVENT>.PtrToStruct(_msg.eventStructPtr, out device_failure_evt);
251  handleFailedDevice(ref device_failure_evt);
252  break;
253 
254  case eLeapEventType.eLeapEventType_Policy:
255  LEAP_POLICY_EVENT policy_evt;
256  StructMarshal<LEAP_POLICY_EVENT>.PtrToStruct(_msg.eventStructPtr, out policy_evt);
257  handlePolicyChange(ref policy_evt);
258  break;
259 
260  case eLeapEventType.eLeapEventType_Tracking:
261  LEAP_TRACKING_EVENT tracking_evt;
262  StructMarshal<LEAP_TRACKING_EVENT>.PtrToStruct(_msg.eventStructPtr, out tracking_evt);
263  handleTrackingMessage(ref tracking_evt);
264  break;
265  case eLeapEventType.eLeapEventType_LogEvent:
266  LEAP_LOG_EVENT log_evt;
267  StructMarshal<LEAP_LOG_EVENT>.PtrToStruct(_msg.eventStructPtr, out log_evt);
268  reportLogMessage(ref log_evt);
269  break;
270  case eLeapEventType.eLeapEventType_DeviceLost:
271  LEAP_DEVICE_EVENT device_lost_evt;
272  StructMarshal<LEAP_DEVICE_EVENT>.PtrToStruct(_msg.eventStructPtr, out device_lost_evt);
273  handleLostDevice(ref device_lost_evt);
274  break;
275  case eLeapEventType.eLeapEventType_ConfigChange:
276  LEAP_CONFIG_CHANGE_EVENT config_change_evt;
277  StructMarshal<LEAP_CONFIG_CHANGE_EVENT>.PtrToStruct(_msg.eventStructPtr, out config_change_evt);
278  handleConfigChange(ref config_change_evt);
279  break;
280  case eLeapEventType.eLeapEventType_ConfigResponse:
281  handleConfigResponse(ref _msg);
282  break;
283  case eLeapEventType.eLeapEventType_DroppedFrame:
284  LEAP_DROPPED_FRAME_EVENT dropped_frame_evt;
285  StructMarshal<LEAP_DROPPED_FRAME_EVENT>.PtrToStruct(_msg.eventStructPtr, out dropped_frame_evt);
286  handleDroppedFrame(ref dropped_frame_evt);
287  break;
288  case eLeapEventType.eLeapEventType_Image:
289  LEAP_IMAGE_EVENT image_evt;
290  StructMarshal<LEAP_IMAGE_EVENT>.PtrToStruct(_msg.eventStructPtr, out image_evt);
291  handleImage(ref image_evt);
292  break;
293  case eLeapEventType.eLeapEventType_PointMappingChange:
294  LEAP_POINT_MAPPING_CHANGE_EVENT point_mapping_change_evt;
295  StructMarshal<LEAP_POINT_MAPPING_CHANGE_EVENT>.PtrToStruct(_msg.eventStructPtr, out point_mapping_change_evt);
296  handlePointMappingChange(ref point_mapping_change_evt);
297  break;
298  case eLeapEventType.eLeapEventType_HeadPose:
299  LEAP_HEAD_POSE_EVENT head_pose_event;
300  StructMarshal<LEAP_HEAD_POSE_EVENT>.PtrToStruct(_msg.eventStructPtr, out head_pose_event);
301  handleHeadPoseChange(ref head_pose_event);
302  break;
303  } //switch on _msg.type
304 
305  if (LeapEndProfilingBlock != null && hasBegunProfilingForThread) {
306  LeapEndProfilingBlock(new EndProfilingBlockArgs(HANDLE_EVENT_PROFILER_BLOCK));
307  }
308  } //while running
309  } catch (Exception e) {
310  Logger.Log("Exception: " + e);
311  _isRunning = false;
312  } finally {
313  if (LeapEndProfilingForThread != null && hasBegunProfilingForThread) {
314  LeapEndProfilingForThread(new EndProfilingForThreadArgs());
315  }
316  }
317  }

377  public void GetInterpolatedLeftRightTransform(Int64 time,
378  Int64 sourceTime,
379  Int64 leftId,
380  Int64 rightId,
381  out LeapTransform leftTransform,
382  out LeapTransform rightTransform) {
383  leftTransform = LeapTransform.Identity;
384  rightTransform = LeapTransform.Identity;
385 
386  UInt64 size = GetInterpolatedFrameSize(time);
387  IntPtr trackingBuffer = Marshal.AllocHGlobal((Int32)size);
388  eLeapRS result = LeapC.InterpolateFrameFromTime(_leapConnection, time, sourceTime, trackingBuffer, size);
389  reportAbnormalResults("LeapC get interpolated frame from time call was ", result);
390 
391  if (result == eLeapRS.eLeapRS_Success) {
392  LEAP_TRACKING_EVENT tracking_evt;
393  StructMarshal<LEAP_TRACKING_EVENT>.PtrToStruct(trackingBuffer, out tracking_evt);
394 
395  int id;
396  LEAP_VECTOR position;
397  LEAP_QUATERNION orientation;
398 
399  long handPtr = tracking_evt.pHands.ToInt64();
400  long idPtr = handPtr + _handIdOffset;
401  long posPtr = handPtr + _handPositionOffset;
402  long rotPtr = handPtr + _handOrientationOffset;
403  int stride = StructMarshal<LEAP_HAND>.Size;
404 
405  for (uint i = tracking_evt.nHands; i-- != 0; idPtr += stride, posPtr += stride, rotPtr += stride) {
406  id = Marshal.ReadInt32(new IntPtr(idPtr));
407  StructMarshal<LEAP_VECTOR>.PtrToStruct(new IntPtr(posPtr), out position);
408  StructMarshal<LEAP_QUATERNION>.PtrToStruct(new IntPtr(rotPtr), out orientation);
409 
410  LeapTransform transform = new LeapTransform(position.ToLeapVector(), orientation.ToLeapQuaternion());
411  if (id == leftId) {
412  leftTransform = transform;
413  } else if (id == rightId) {
414  rightTransform = transform;
415  }
416  }
417  }
418 
419  Marshal.FreeHGlobal(trackingBuffer);
420  }

434  private void handleDevice(ref LEAP_DEVICE_EVENT deviceMsg) {
435  IntPtr deviceHandle = deviceMsg.device.handle;
436  if (deviceHandle == IntPtr.Zero)
437  return;
438 
439  LEAP_DEVICE_INFO deviceInfo = new LEAP_DEVICE_INFO();
440  eLeapRS result;
441 
442  IntPtr device;
443  result = LeapC.OpenDevice(deviceMsg.device, out device);
444  if (result != eLeapRS.eLeapRS_Success)
445  return;
446 
447  deviceInfo.serial = IntPtr.Zero;
448  deviceInfo.size = (uint)Marshal.SizeOf(deviceInfo);
449  result = LeapC.GetDeviceInfo(device, ref deviceInfo); //Query the serial length
450  if (result != eLeapRS.eLeapRS_Success)
451  return;
452 
453  deviceInfo.serial = Marshal.AllocCoTaskMem((int)deviceInfo.serial_length);
454  result = LeapC.GetDeviceInfo(device, ref deviceInfo); //Query the serial
455 
456  if (result == eLeapRS.eLeapRS_Success) {
457  Device apiDevice = new Device(deviceHandle,
458  deviceInfo.h_fov, //radians
459  deviceInfo.v_fov, //radians
460  deviceInfo.range / 1000.0f, //to mm
461  deviceInfo.baseline / 1000.0f, //to mm
462  (Device.DeviceType)deviceInfo.type,
463  (deviceInfo.status == eLeapDeviceStatus.eLeapDeviceStatus_Streaming),
464  Marshal.PtrToStringAnsi(deviceInfo.serial));
465  Marshal.FreeCoTaskMem(deviceInfo.serial);
466  _devices.AddOrUpdate(apiDevice);
467 
468  if (LeapDevice != null) {
469  LeapDevice.DispatchOnContext(this, EventContext, new DeviceEventArgs(apiDevice));
470  }
471  }

527  private void handleConfigResponse(ref LEAP_CONNECTION_MESSAGE configMsg) {
528  LEAP_CONFIG_RESPONSE_EVENT config_response_evt;
529  StructMarshal<LEAP_CONFIG_RESPONSE_EVENT>.PtrToStruct(configMsg.eventStructPtr, out config_response_evt);
530  string config_key = "";
531  _configRequests.TryGetValue(config_response_evt.requestId, out config_key);
532  if (config_key != null)
533  _configRequests.Remove(config_response_evt.requestId);
534 
535  Config.ValueType dataType;
536  object value;
537  uint requestId = config_response_evt.requestId;
538  if (config_response_evt.value.type != eLeapValueType.eLeapValueType_String) {
539  switch (config_response_evt.value.type) {
540  case eLeapValueType.eLeapValueType_Boolean:
541  dataType = Config.ValueType.TYPE_BOOLEAN;
542  value = config_response_evt.value.boolValue;
543  break;
544  case eLeapValueType.eLeapValueType_Int32:
545  dataType = Config.ValueType.TYPE_INT32;
546  value = config_response_evt.value.intValue;
547  break;
548  case eLeapValueType.eLeapValueType_Float:
549  dataType = Config.ValueType.TYPE_FLOAT;
550  value = config_response_evt.value.floatValue;
551  break;
552  default:
553  dataType = Config.ValueType.TYPE_UNKNOWN;
554  value = new object();
555  break;
556  }
557  } else {
558  LEAP_CONFIG_RESPONSE_EVENT_WITH_REF_TYPE config_ref_value;
559  StructMarshal<LEAP_CONFIG_RESPONSE_EVENT_WITH_REF_TYPE>.PtrToStruct(configMsg.eventStructPtr, out config_ref_value);
560  dataType = Config.ValueType.TYPE_STRING;
561  value = config_ref_value.value.stringValue;
562  }
563  SetConfigResponseEventArgs args = new SetConfigResponseEventArgs(config_key, dataType, value, requestId);
564 
565  if (LeapConfigResponse != null) {
566  LeapConfigResponse.DispatchOnContext(this, EventContext, args);
567  }

820  public void GetPointMapping(ref PointMapping pm) {
821  UInt64 size = 0;
822  IntPtr buffer = IntPtr.Zero;
823  while (true) {
824  eLeapRS result = LeapC.GetPointMapping(_leapConnection, buffer, ref size);
825  if (result == eLeapRS.eLeapRS_InsufficientBuffer) {
826  if (buffer != IntPtr.Zero)
827  Marshal.FreeHGlobal(buffer);
828  buffer = Marshal.AllocHGlobal((Int32)size);
829  continue;
830  }
831  reportAbnormalResults("LeapC get point mapping call was ", result);
832  if (result != eLeapRS.eLeapRS_Success) {
833  pm.points = null;
834  pm.ids = null;
835  return;
836  }
837  break;
838  }
839  LEAP_POINT_MAPPING pmi;
840  StructMarshal<LEAP_POINT_MAPPING>.PtrToStruct(buffer, out pmi);
841  Int32 nPoints = (Int32)pmi.nPoints;
842 
843  pm.frameId = pmi.frame_id;
844  pm.timestamp = pmi.timestamp;
845  pm.points = new Vector[nPoints];
846  pm.ids = new UInt32[nPoints];
847 
848  float[] points = new float[3 * nPoints];
849  Int32[] ids = new Int32[nPoints];
850  Marshal.Copy(pmi.points, points, 0, 3 * nPoints);
851  Marshal.Copy(pmi.ids, ids, 0, nPoints);
852 
853  int j = 0;
854  for (int i = 0; i < nPoints; i++) {
855  pm.points[i].x = points[j++];
856  pm.points[i].y = points[j++];
857  pm.points[i].z = points[j++];
858  pm.ids[i] = unchecked((UInt32)ids[i]);
859  }
860  Marshal.FreeHGlobal(buffer);
861  }
