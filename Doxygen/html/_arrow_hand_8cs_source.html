107  float distanceToNockPosition = Vector3.Distance( transform.parent.position, bow.nockTransform.position );
108 
109  // If there's an arrow spawned in the hand and it's not nocked yet
110  if ( !nocked )
111  {
112  // If we're close enough to nock position that we want to start arrow rotation lerp, do so
113  if ( distanceToNockPosition < rotationLerpThreshold )
114  {
115  float lerp = Util.RemapNumber( distanceToNockPosition, rotationLerpThreshold, lerpCompleteDistance, 0, 1 );
116 
117  arrowNockTransform.rotation = Quaternion.Lerp( arrowNockTransform.parent.rotation, bow.nockRestTransform.rotation, lerp );
118  }
119  else // Not close enough for rotation lerp, reset rotation
120  {
121  arrowNockTransform.localRotation = Quaternion.identity;
122  }
123 
124  // If we're close enough to the nock position that we want to start arrow position lerp, do so
125  if ( distanceToNockPosition < positionLerpThreshold )
126  {
127  float posLerp = Util.RemapNumber( distanceToNockPosition, positionLerpThreshold, lerpCompleteDistance, 0, 1 );
128 
129  posLerp = Mathf.Clamp( posLerp, 0f, 1f );
130 
131  arrowNockTransform.position = Vector3.Lerp( arrowNockTransform.parent.position, bow.nockRestTransform.position, posLerp );
132  }
133  else // Not close enough for position lerp, reset position
134  {
135  arrowNockTransform.position = arrowNockTransform.parent.position;
136  }
137 
138 
139  // Give a haptic tick when lerp is visually complete
140  if ( distanceToNockPosition < lerpCompleteDistance )
141  {
142  if ( !arrowLerpComplete )
143  {
144  arrowLerpComplete = true;
145  hand.TriggerHapticPulse( 500 );
146  }
147  }
148  else
149  {
150  if ( arrowLerpComplete )
151  {
152  arrowLerpComplete = false;
153  }
154  }
155 
156  // Allow nocking the arrow when controller is close enough
157  if ( distanceToNockPosition < nockDistance )
158  {
159  if ( !inNockRange )
160  {
161  inNockRange = true;
162  bow.ArrowInPosition();
163  }
164  }
165  else
166  {
167  if ( inNockRange )
168  {
169  inNockRange = false;
170  }
171  }
172 
173  GrabTypes bestGrab = hand.GetBestGrabbingType(GrabTypes.Pinch, true);
174 
175  // If arrow is close enough to the nock position and we're pressing the trigger, and we're not nocked yet, Nock
176  if ( ( distanceToNockPosition < nockDistance ) && bestGrab != GrabTypes.None && !nocked )
177  {
178  if ( currentArrow == null )
179  {
180  currentArrow = InstantiateArrow();
181  }
182 
183  nocked = true;
184  nockedWithType = bestGrab;
185  bow.StartNock( this );
186  hand.HoverLock( GetComponent<Interactable>() );
187  allowTeleport.teleportAllowed = false;
188  currentArrow.transform.parent = bow.nockTransform;
189  Util.ResetTransform( currentArrow.transform );
190  Util.ResetTransform( arrowNockTransform );
191  }
192  }
193 
194 
195  // If arrow is nocked, and we release the trigger
196  if ( nocked && hand.IsGrabbingWithType(nockedWithType) == false )
197  {
198  if ( bow.pulled ) // If bow is pulled back far enough, fire arrow, otherwise reset arrow in arrowhand
199  {
200  FireArrow();
201  }
202  else
203  {
204  arrowNockTransform.rotation = currentArrow.transform.rotation;
205  currentArrow.transform.parent = arrowNockTransform;
206  Util.ResetTransform( currentArrow.transform );
207  nocked = false;
208  nockedWithType = GrabTypes.None;
209  bow.ReleaseNock();
210  hand.HoverUnlock( GetComponent<Interactable>() );
211  allowTeleport.teleportAllowed = true;
212  }
213 
214  bow.StartRotationLerp(); // Arrow is releasing from the bow, tell the bow to lerp back to controller rotation
215  }
