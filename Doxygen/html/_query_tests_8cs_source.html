515  public class QueryArg {
516  private int[] _array;
517  private int _count;
518 
519  public QueryArg(List<int> values, int capacity) {
520  _array = new int[capacity];
521  values.CopyTo(_array);
522  _count = values.Count;
523  }
524 
525  public Query<int> ToQuery() {
526  int[] copy = new int[_array.Length];
527  _array.CopyTo(copy, 0);
528  return new Query<int>(copy, _count);
529  }
530 
531  public List<int> ToList() {
532  return new List<int>(_array.Take(_count));
533  }
534 
535  public override string ToString() {
536  return _array.Length + " : " + Utils.ToArrayString(_array.Take(_count));
537  }
