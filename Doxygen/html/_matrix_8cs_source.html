231  public void SetRotation(Vector axis, float angleRadians) {
232  Vector n = axis.Normalized;
233  float s = (float)Math.Sin(angleRadians);
234  float c = (float)Math.Cos(angleRadians);
235  float C = (1 - c);
236 
237  xBasis = new Vector(n[0] * n[0] * C + c, n[0] * n[1] * C - n[2] * s, n[0] * n[2] * C + n[1] * s);
238  yBasis = new Vector(n[1] * n[0] * C + n[2] * s, n[1] * n[1] * C + c, n[1] * n[2] * C - n[0] * s);
239  zBasis = new Vector(n[2] * n[0] * C - n[1] * s, n[2] * n[1] * C + n[0] * s, n[2] * n[2] * C + c);
240  }
