84  protected virtual void UpdateHandRepresentations(Dictionary<int, HandRepresentation> all_hand_reps, ModelType modelType, Frame frame) {
85  for (int i = 0; i < frame.Hands.Count; i++) {
86  var curHand = frame.Hands[i];
87  HandRepresentation rep;
88  if (!all_hand_reps.TryGetValue(curHand.Id, out rep)) {
89  rep = MakeHandRepresentation(curHand, modelType);
90  all_hand_reps.Add(curHand.Id, rep);
91  }
92  if (rep != null) {
93  rep.IsMarked = true;
94  rep.UpdateRepresentation(curHand);
95  rep.LastUpdatedTime = (int)frame.Timestamp;
96  }
97  }
98 
100  HandRepresentation toBeDeleted = null;
101  for (var it = all_hand_reps.GetEnumerator(); it.MoveNext();) {
102  var r = it.Current;
103  if (r.Value != null) {
104  if (r.Value.IsMarked) {
105  r.Value.IsMarked = false;
106  }
107  else {
109  //Debug.Log("Finishing");
110  toBeDeleted = r.Value;
111  }
112  }
113  }
116  if (toBeDeleted != null) {
117  all_hand_reps.Remove(toBeDeleted.HandID);
118  toBeDeleted.Finish();
119  }
120  }

194  for (int i = 0; i < modelList.Count; i++) {
195  if (modelList[i].HandModelType == modelType && modelList[i].Handedness == chirality) {

204  if (modelsCheckedOut[i].HandModelType == modelType && modelsCheckedOut[i].Handedness == chirality) {
205  HandModelBase modelToSpawn = modelsCheckedOut[i];
206  HandModelBase spawnedModel = GameObject.Instantiate(modelToSpawn);
207  spawnedModel.transform.parent = _handModelManager.transform;
208  _handModelManager.modelGroupMapping.Add(spawnedModel, this);

213  }

227  public void ReturnToPool(HandModelBase model) {
228  ModelGroup modelGroup;
229  bool groupFound = modelGroupMapping.TryGetValue(model, out modelGroup);
230  Assert.IsTrue(groupFound);
231  //First see if there is another active Representation that can use this model
232  for (int i = 0; i < activeHandReps.Count; i++) {
233  HandRepresentation rep = activeHandReps[i];
234  if (rep.RepChirality == model.Handedness && rep.RepType == model.HandModelType) {
235  bool modelFromGroupFound = false;
236  if (rep.handModels != null) {
237  //And that Represention does not contain a model from this model's modelGroup
238  for (int j = 0; j < modelGroup.modelsCheckedOut.Count; j++) {
239  HandModelBase modelToCompare = modelGroup.modelsCheckedOut[j];
240  for (int k = 0; k < rep.handModels.Count; k++) {
241  if (rep.handModels[k] == modelToCompare) {
242  modelFromGroupFound = true;
243  }
244  }
245  }
246  }
247  if (!modelFromGroupFound) {
248  rep.AddModel(model);
249  modelToHandRepMapping[model] = rep;
250  return;
251  }
252  }
253  }
254  //Otherwise return to pool
255  modelGroup.ReturnToGroup(model);
256  }

267  public HandRepresentation MakeHandRepresentation(Hand hand, ModelType modelType) {
268  Chirality handChirality = hand.IsRight ? Chirality.Right : Chirality.Left;
269  HandRepresentation handRep = new HandRepresentation(this, hand, handChirality, modelType);
270  for (int i = 0; i < ModelPool.Count; i++) {
271  ModelGroup group = ModelPool[i];
272  if (group.IsEnabled) {
273  HandModelBase model = group.TryGetModel(handChirality, modelType);
274  if (model != null) {
275  handRep.AddModel(model);
276  if (!modelToHandRepMapping.ContainsKey(model)) {
277  model.group = group;
278  modelToHandRepMapping.Add(model, handRep);
279  }
280  }
281  }
282  }
283  activeHandReps.Add(handRep);
284  return handRep;
285  }
