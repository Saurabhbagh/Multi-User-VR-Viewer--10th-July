78  // Hooked up via PhysicsCallbacks in Start(), this method will run after

107  private void updateHandles() {
108  switch (_toolState) {
109  case ToolState.Idle:
110  // Find the closest handle to any InteractionHand.
111  TransformHandle closestHandleToAnyHand = null;
112  float closestHandleDist = float.PositiveInfinity;
113  foreach (var intController in interactionManager.interactionControllers
114  .Query()
115  .Where(controller => controller.isTracked)) {
116  if (!intController.isPrimaryHovering) continue;
117  TransformHandle testHandle = intController.primaryHoveredObject
118  .gameObject
119  .GetComponent<TransformHandle>();
120 
121  if (testHandle == null || !_transformHandles.Contains(testHandle)) continue;
122 
123  float testDist = intController.primaryHoverDistance;
124  if (testDist < closestHandleDist) {
125  closestHandleToAnyHand = testHandle;
126  closestHandleDist = testDist;
127  }
128  }
129 
130  // While idle, only show the closest handle to any hand, hide other handles.
131  foreach (var handle in _transformHandles) {
132  if (closestHandleToAnyHand != null && handle == closestHandleToAnyHand) {
133  handle.EnsureVisible();
134  }
135  else {
136  handle.EnsureHidden();
137  }
138  }
139  break;
140 
141  case ToolState.Translating:
142  // While translating, show all translation handles except the other handle
143  // on the same axis, and hide rotation handles.
144  foreach (var handle in _transformHandles) {
145  if (handle is TransformTranslationHandle) {
146  var translateHandle = handle as TransformTranslationHandle;
147 
148  if (!_activeHandles.Contains(translateHandle)
149  && _activeTranslationAxes.Contains(translateHandle.axis)) {
150  handle.EnsureHidden();
151  }
152  else {
153  handle.EnsureVisible();
154  }
155  }
156  else {
157  handle.EnsureHidden();
158  }
159  }
160  break;
161 
162  case ToolState.Rotating:
163  // While rotating, only show the active rotating handle.
164  foreach (var handle in _transformHandles) {
165  if (_activeHandles.Contains(handle)) {
166  handle.EnsureVisible();
167  }
168  else {
169  handle.EnsureHidden();
170  }
171  }
172  break;
173  }
174  }
