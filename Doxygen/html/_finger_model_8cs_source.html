167  public float GetFingerJointStretchMecanim(int joint_type) {
168  // The successive actions of local rotations on a vector yield the global rotation,
169  // so the inverse of the parent rotation appears on the left.
170  Quaternion jointRotation = Quaternion.identity;
171  if (finger_ != null) {
172  jointRotation = Quaternion.Inverse(finger_.Bone((Bone.BoneType)(joint_type)).Rotation.ToQuaternion())
173  * finger_.Bone ((Bone.BoneType)(joint_type + 1)).Rotation.ToQuaternion();
174  } else if (bones [joint_type] && bones [joint_type + 1]) {
175  jointRotation = Quaternion.Inverse (GetBoneRotation (joint_type)) * GetBoneRotation (joint_type + 1);
176  }
177  // Stretch is a rotation around the X axis of the base bone
178  // Positive stretch opens joints
179  float stretchAngle = -jointRotation.eulerAngles.x;
180  if (stretchAngle <= -180f) {
181  stretchAngle += 360f;
182  }
183  // NOTE: eulerAngles range is [0, 360) so stretchAngle > +180f will not occur.
184  return stretchAngle;
185  }

193  public float GetFingerJointSpreadMecanim() {
194  // The successive actions of local rotations on a vector yield the global rotation,
195  // so the inverse of the parent rotation appears on the left.
196  Quaternion jointRotation = Quaternion.identity;
197  if (finger_ != null) {
198  jointRotation = Quaternion.Inverse (finger_.Bone ((Bone.BoneType)(0)).Rotation.ToQuaternion())
199  * finger_.Bone ((Bone.BoneType)(1)).Rotation.ToQuaternion();
200  } else if (bones [0] && bones [1]) {
201  jointRotation = Quaternion.Inverse (GetBoneRotation (0)) * GetBoneRotation (1);
202  }
203  // Spread is a rotation around the Y axis of the base bone when joint_type = 0
204  float spreadAngle = 0f;
205  Finger.FingerType fType = fingerType;
206  if (finger_ != null) {

208  }

212  spreadAngle = jointRotation.eulerAngles.y;
