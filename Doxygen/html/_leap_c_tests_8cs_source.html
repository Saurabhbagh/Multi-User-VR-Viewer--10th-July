195  Logger.Log("Test : " + testTime);
196  Logger.Log("NextF : " + firstAfter.info.timestamp);
197 
198  result = LeapC.GetFrameSize(connHandle, testTime, out size);
199  Logger.Log("Size: " + size);
200  Assert.AreEqual(eLeapRS.eLeapRS_Success, result, "Frame size call succeeded " + result.indexOf());
201

205 
206  LEAP_TRACKING_EVENT tracking_evt;
