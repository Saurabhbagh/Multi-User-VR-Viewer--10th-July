136  private bool updateBehaviour(IInteractionBehaviour behaviour, Hand hand, GraspUpdateMode graspMode, bool ignoreTemporal = false) {
137  using (new ProfilerSample("Update Individual Grab Classifier", behaviour.gameObject)) {
138  // Ensure a classifier exists for this Interaction Behaviour.
139  GrabClassifierHeuristics.GrabClassifier classifier;
140  if (!_classifiers.TryGetValue(behaviour, out classifier)) {
141  classifier = new GrabClassifierHeuristics.GrabClassifier(behaviour.gameObject);
142  _classifiers.Add(behaviour, classifier);
143  }
144 
145  // Do the actual grab classification logic.
146  FillClassifier(behaviour, hand, ref classifier);
147  GrabClassifierHeuristics.UpdateClassifier(classifier, _scaledGrabParams,
148  ref _collidingCandidates,
149  ref _numberOfColliders,
150  ignoreTemporal);
151 
152  // Determine whether there was a state change.
153  bool didStateChange = false;
154  if (!classifier.prevThisControllerGrabbing && classifier.isThisControllerGrabbing
155  && graspMode == GraspUpdateMode.BeginGrasp) {
156  didStateChange = true;
157 
158  classifier.prevThisControllerGrabbing = classifier.isThisControllerGrabbing;
159  }
160  else if (classifier.prevThisControllerGrabbing && !classifier.isThisControllerGrabbing
161  && interactionHand.graspedObject == behaviour && graspMode == GraspUpdateMode.ReleaseGrasp) {
162  didStateChange = true;
163 
164  classifier.coolDownProgress = 0f;
165  classifier.prevThisControllerGrabbing = classifier.isThisControllerGrabbing;
166  }
167 
168  return didStateChange;
169  }
