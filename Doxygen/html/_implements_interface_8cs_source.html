60  public UnityObject FindImplementer(SerializedProperty property, UnityObject fromObj) {
61 
62  // Don't use fieldInfo here because it is very convenient for it to be left null
63  // on the CombinablePropertyAttribute when dealing with generic-type situations
64  // where the ImplementsInterface class has to be constructed manually in a custom
65  // editor. (Specific case: StreamConnectorEditor.cs)
66 
67  bool isTypeDefinitelyIncompatible;
68  {
69  isTypeDefinitelyIncompatible = !this.type.IsAssignableFrom(fromObj.GetType());
70 
71  // We have to make an exception when a GameObject is dragged into a field whose
72  // type is a Component, because it's expected that we can use GetComponent in
73  // that case. (We might still fail later if the component isn't found.)
74  var objIsGameObject = fromObj.GetType() == typeof(GameObject);
75  if (objIsGameObject) {
76  isTypeDefinitelyIncompatible = false;
77  }
78 
79  // We also make an exception when a Component is dragged into a field that
80  // doesn't directly satisfy the interface because it might have other Components
81  // on its GameObject that _do_ satisfy the field, again via GetComponent.
82  var objIsComponent = typeof(Component).IsAssignableFrom(fromObj.GetType());
83  if (objIsComponent) {
84  isTypeDefinitelyIncompatible = false;
85  }
86 
87  // However, you can't assign a ScriptableObject to a field expecting a Component,
88  // or vice-versa, no matter what.
89 
90  var fieldIsScriptableObject
91  = fieldInfo.FieldType.IsAssignableFrom(typeof(ScriptableObject));
92  if (fieldIsScriptableObject && (objIsComponent || objIsGameObject)) {
93  isTypeDefinitelyIncompatible = true;
94  }
95 
96  var fieldTakesComponent
97  = typeof(Component).IsAssignableFrom(fieldInfo.FieldType);
98  if (fieldTakesComponent && (!objIsComponent && !objIsGameObject)) {
99  isTypeDefinitelyIncompatible = true;
100  }
101  }
102  if (isTypeDefinitelyIncompatible) {
103  return null;
104  }
105 
106  if (fromObj.GetType().ImplementsInterface(type)) {
107  // All good! This object reference implements the interface.
108  return fromObj;
109  }
110  else {
111  UnityObject implementingObject;
112 
113  if (fromObj is GameObject) {
114  fromObj = (fromObj as GameObject).transform;
115  }
116 
117  if (fromObj is Component) {
118  // If the object is a Component, first search the rest of the GameObject
119  // for a component that implements the interface. If found, assign it instead,
120  // otherwise null out the property.
121  implementingObject = (fromObj as Component)
122  .GetComponents<Component>()
123  .Query()
124  .Where(c => c.GetType().ImplementsInterface(type))
125  .FirstOrDefault();
126  }
127  else {
128  // If the object is not a Component, just null out the property.
129  implementingObject = null;
130  }
131 
132  return implementingObject;
133  }
134  }
