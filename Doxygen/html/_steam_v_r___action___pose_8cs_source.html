596  protected bool GetChanged()
597  {
598  if (Vector3.Distance(localPosition, lastLocalPosition) > changeTolerance)
599  return true;
600  else if (Mathf.Abs(Quaternion.Angle(localRotation, lastLocalRotation)) > changeTolerance)

602  else if (Vector3.Distance(velocity, lastVelocity) > changeTolerance)
603  return true;
604  else if (Vector3.Distance(angularVelocity, lastAngularVelocity) > changeTolerance)
