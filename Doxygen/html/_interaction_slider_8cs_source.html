365  private void calculateSliderLimits() {
366  if (transform.parent != null) {
367  parent = transform.parent.GetComponent<RectTransform>();
368 
369  if (overrideRectLimits) return;
370 
371  if (parent != null) {
372  if (parent.rect.width < 0f || parent.rect.height < 0f) {
373  Debug.LogError("Parent Rectangle dimensions negative; can't set slider boundaries!", parent.gameObject);
374  enabled = false;
375  } else {
376  var self = transform.GetComponent<RectTransform>();
377  if (self != null) {
378  horizontalSlideLimits = new Vector2(parent.rect.xMin - transform.localPosition.x + self.rect.width / 2F, parent.rect.xMax - transform.localPosition.x - self.rect.width / 2F);
379  if (horizontalSlideLimits.x > horizontalSlideLimits.y) {
380  horizontalSlideLimits = new Vector2(0F, 0F);
381  }
382  if (Mathf.Abs(horizontalSlideLimits.x) < 0.0001F) {
383  horizontalSlideLimits.x = 0F;
384  }
385  if (Mathf.Abs(horizontalSlideLimits.y) < 0.0001F) {
386  horizontalSlideLimits.y = 0F;
387  }
388 
389  verticalSlideLimits = new Vector2(parent.rect.yMin - transform.localPosition.y + self.rect.height / 2F, parent.rect.yMax - transform.localPosition.y - self.rect.height / 2F);
390  if (verticalSlideLimits.x > verticalSlideLimits.y) {
391  verticalSlideLimits = new Vector2(0F, 0F);
392  }
393  if (Mathf.Abs(verticalSlideLimits.x) < 0.0001F) {
394  verticalSlideLimits.x = 0F;
395  }
396  if (Mathf.Abs(verticalSlideLimits.y) < 0.0001F) {
397  verticalSlideLimits.y = 0F;
398  }
399  } else {
400  horizontalSlideLimits = new Vector2(parent.rect.xMin - transform.localPosition.x, parent.rect.xMax - transform.localPosition.x);
401  verticalSlideLimits = new Vector2(parent.rect.yMin - transform.localPosition.y, parent.rect.yMax - transform.localPosition.y);
402  }
403  }
404  }
405  }

408  protected override Vector3 constrainDepressedLocalPosition(Vector3 desiredOffset) {
409  Vector3 unSnappedPosition =
410  new Vector3(Mathf.Clamp((localPhysicsPosition.x + desiredOffset.x), initialLocalPosition.x + horizontalSlideLimits.x, initialLocalPosition.x + horizontalSlideLimits.y),
411  Mathf.Clamp((localPhysicsPosition.y + desiredOffset.y), initialLocalPosition.y + verticalSlideLimits.x, initialLocalPosition.y + verticalSlideLimits.y),
412  (localPhysicsPosition.z + desiredOffset.z));
413 
414  float hSliderPercent = Mathf.InverseLerp(initialLocalPosition.x + horizontalSlideLimits.x, initialLocalPosition.x + horizontalSlideLimits.y, unSnappedPosition.x);
415  if (horizontalSteps > 0) {
416  hSliderPercent = Mathf.Round(hSliderPercent * (horizontalSteps)) / (horizontalSteps);
417  }
418 
419  float vSliderPercent = Mathf.InverseLerp(initialLocalPosition.y + verticalSlideLimits.x, initialLocalPosition.y + verticalSlideLimits.y, unSnappedPosition.y);
420  if (verticalSteps > 0) {
421  vSliderPercent = Mathf.Round(vSliderPercent * (verticalSteps)) / (verticalSteps);
422  }
423 
424  return new Vector3(Mathf.Lerp(initialLocalPosition.x + horizontalSlideLimits.x, initialLocalPosition.x + horizontalSlideLimits.y, hSliderPercent),
425  Mathf.Lerp(initialLocalPosition.y + verticalSlideLimits.x, initialLocalPosition.y + verticalSlideLimits.y, vSliderPercent),
426  (localPhysicsPosition.z + desiredOffset.z));
427  }
