467  protected IEnumerator DoRangeOfMotionBlend(EVRSkeletalMotionRange oldRangeOfMotion, EVRSkeletalMotionRange newRangeOfMotion, float overTime)
468  {
469  float startTime = Time.time;
470  float endTime = startTime + overTime;
471 
472  Vector3[] oldBonePositions;
473  Quaternion[] oldBoneRotations;
474 
475  Vector3[] newBonePositions;
476  Quaternion[] newBoneRotations;
477 
478  while (Time.time < endTime)
479  {
480  yield return null;
481  float lerp = (Time.time - startTime) / overTime;
482 
483  if (skeletonBlend > 0)
484  {
485  skeletonAction.SetRangeOfMotion(oldRangeOfMotion);
486  skeletonAction.UpdateValueWithoutEvents();
487  oldBonePositions = (Vector3[])GetBonePositions().Clone();
488  oldBoneRotations = (Quaternion[])GetBoneRotations().Clone();
489 
490  skeletonAction.SetRangeOfMotion(newRangeOfMotion);
491  skeletonAction.UpdateValueWithoutEvents();
492  newBonePositions = GetBonePositions();
493  newBoneRotations = GetBoneRotations();
494 
495  for (int boneIndex = 0; boneIndex < bones.Length; boneIndex++)
496  {
497  if (bones[boneIndex] == null)
498  continue;
499 
500  if (SteamVR_Utils.IsValid(newBoneRotations[boneIndex]) == false || SteamVR_Utils.IsValid(oldBoneRotations[boneIndex]) == false)
501  {
502  continue;
503  }
504 
505  Vector3 blendedRangeOfMotionPosition = Vector3.Lerp(oldBonePositions[boneIndex], newBonePositions[boneIndex], lerp);
506  Quaternion blendedRangeOfMotionRotation = Quaternion.Lerp(oldBoneRotations[boneIndex], newBoneRotations[boneIndex], lerp);
507 
508  if (skeletonBlend < 1)
509  {
510  if (blendPoser != null)
511  {
512  SetBonePosition(boneIndex, Vector3.Lerp(blendSnapshot.bonePositions[boneIndex], blendedRangeOfMotionPosition, skeletonBlend));
513  SetBoneRotation(boneIndex, Quaternion.Lerp(GetBlendPoseForBone(boneIndex, blendedRangeOfMotionRotation), blendedRangeOfMotionRotation, skeletonBlend));
514  }
515  else
516  {
517  SetBonePosition(boneIndex, Vector3.Lerp(bones[boneIndex].localPosition, blendedRangeOfMotionPosition, skeletonBlend));
518  SetBoneRotation(boneIndex, Quaternion.Lerp(bones[boneIndex].localRotation, blendedRangeOfMotionRotation, skeletonBlend));
519  }
520  }
521  else
522  {
523  SetBonePosition(boneIndex, blendedRangeOfMotionPosition);
524  SetBoneRotation(boneIndex, blendedRangeOfMotionRotation);
525  }
526  }
527  }
528 
529  if (onBoneTransformsUpdated != null)
530  onBoneTransformsUpdated.Invoke(this, inputSource);
531  if (onBoneTransformsUpdatedEvent != null)
532  onBoneTransformsUpdatedEvent.Invoke(this, inputSource);
533 
534  }
535 
536  rangeOfMotionBlendRoutine = null;
537  }

545  protected virtual void UpdateSkeletonTransforms()
546  {
547  Vector3[] bonePositions = GetBonePositions();
548  Quaternion[] boneRotations = GetBoneRotations();
549 
550  if (skeletonBlend <= 0)
551  {
552  if (blendPoser != null)
553  {
554  SteamVR_Skeleton_Pose_Hand mainPose = blendPoser.skeletonMainPose.GetHand(inputSource);
555  for (int boneIndex = 0; boneIndex < bones.Length; boneIndex++)
556  {
557  if (bones[boneIndex] == null)
558  continue;
559 
560  if ((boneIndex == SteamVR_Skeleton_JointIndexes.wrist && mainPose.ignoreWristPoseData) ||
561  (boneIndex == SteamVR_Skeleton_JointIndexes.root && mainPose.ignoreRootPoseData))
562  {
563  SetBonePosition(boneIndex, bonePositions[boneIndex]);
564  SetBoneRotation(boneIndex, boneRotations[boneIndex]);
565  }
566  else
567  {
568  Quaternion poseRotation = GetBlendPoseForBone(boneIndex, boneRotations[boneIndex]);
569 
570  SetBonePosition(boneIndex, blendSnapshot.bonePositions[boneIndex]);
571  SetBoneRotation(boneIndex, poseRotation);
572  }
573  }
574  }
575  }
576  else if (skeletonBlend >= 1)
577  {
578  for (int boneIndex = 0; boneIndex < bones.Length; boneIndex++)
579  {
580  if (bones[boneIndex] == null)
581  continue;
582 
583  SetBonePosition(boneIndex, bonePositions[boneIndex]);
584  SetBoneRotation(boneIndex, boneRotations[boneIndex]);
585  }
586  }
587  else
588  {
589  for (int boneIndex = 0; boneIndex < bones.Length; boneIndex++)
590  {
591  if (bones[boneIndex] == null)
592  continue;
593 
594  if (blendPoser != null)
595  {
596  SteamVR_Skeleton_Pose_Hand mainPose = blendPoser.skeletonMainPose.GetHand(inputSource);
597 
598  if ((boneIndex == SteamVR_Skeleton_JointIndexes.wrist && mainPose.ignoreWristPoseData) ||
599  (boneIndex == SteamVR_Skeleton_JointIndexes.root && mainPose.ignoreRootPoseData))
600  {
601  SetBonePosition(boneIndex, bonePositions[boneIndex]);
602  SetBoneRotation(boneIndex, boneRotations[boneIndex]);
603  }
604  else
605  {
606  Quaternion poseRotation = GetBlendPoseForBone(boneIndex, boneRotations[boneIndex]);
607 
608  SetBonePosition(boneIndex, Vector3.Lerp(blendSnapshot.bonePositions[boneIndex], bonePositions[boneIndex], skeletonBlend));
609  SetBoneRotation(boneIndex, Quaternion.Lerp(poseRotation, boneRotations[boneIndex], skeletonBlend));

722  protected virtual void UpdatePose()
723  {
724  if (skeletonAction == null)
725  return;
726 
727  Vector3 skeletonPosition = skeletonAction.GetLocalPosition();
728  Quaternion skeletonRotation = skeletonAction.GetLocalRotation();
729  if (origin == null)
730  {
731  if (this.transform.parent != null)
732  {
733  skeletonPosition = this.transform.parent.TransformPoint(skeletonPosition);
734  skeletonRotation = this.transform.parent.rotation * skeletonRotation;
735  }
736  }
737  else
738  {
739  skeletonPosition = origin.TransformPoint(skeletonPosition);
740  skeletonRotation = origin.rotation * skeletonRotation;
741  }
742 
743  if (skeletonAction.poseChanged)
744  {
745  if (onTransformChanged != null)
746  onTransformChanged.Invoke(this, inputSource);
747  if (onTransformChangedEvent != null)
748  onTransformChangedEvent.Invoke(this, inputSource);
749  }

766  {

775  float increment = 100;
