177  public virtual void GetReleaseVelocities(Hand hand, out Vector3 velocity, out Vector3 angularVelocity)
178  {
179  if (hand.noSteamVRFallbackCamera && releaseVelocityStyle != ReleaseStyle.NoChange)
180  releaseVelocityStyle = ReleaseStyle.ShortEstimation; // only type that works with fallback hand is short estimation.
181 
182  switch (releaseVelocityStyle)
183  {
184  case ReleaseStyle.ShortEstimation:
185  velocityEstimator.FinishEstimatingVelocity();
186  velocity = velocityEstimator.GetVelocityEstimate();
187  angularVelocity = velocityEstimator.GetAngularVelocityEstimate();
188  break;
189  case ReleaseStyle.AdvancedEstimation:
190  hand.GetEstimatedPeakVelocities(out velocity, out angularVelocity);
191  break;
192  case ReleaseStyle.GetFromHand:
193  velocity = hand.GetTrackedObjectVelocity(releaseVelocityTimeOffset);
194  angularVelocity = hand.GetTrackedObjectAngularVelocity(releaseVelocityTimeOffset);
195  break;
196  default:
197  case ReleaseStyle.NoChange:
198  velocity = rigidbody.velocity;
199  angularVelocity = rigidbody.angularVelocity;
200  break;
201  }
202 
203  if (releaseVelocityStyle != ReleaseStyle.NoChange)
204  velocity *= scaleReleaseVelocity;
205  }
