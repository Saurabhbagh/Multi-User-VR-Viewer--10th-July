230  public void ShowMenu()
231  {
232  var overlay = SteamVR_Overlay.instance;
233  if (overlay == null)
234  return;
235 
236  var texture = overlay.texture as RenderTexture;
237  if (texture == null)
238  {
239  Debug.LogError("<b>[SteamVR]</b> Menu requires overlay texture to be a render texture.");
240  return;
241  }
242 
243  SaveCursorState();
244 
245  Cursor.visible = true;
246  Cursor.lockState = CursorLockMode.None;
247 
248  this.overlay = overlay;
249  uvOffset = overlay.uvOffset;
250  distance = overlay.distance;
251 
252  // If an existing camera is rendering into the overlay texture, we need
253  // to temporarily disable it to keep it from clearing the texture on us.
254  var cameras = Object.FindObjectsOfType(typeof(Camera)) as Camera[];
255  foreach (var cam in cameras)
256  {
257  if (cam.enabled && cam.targetTexture == texture)
258  {
259  overlayCam = cam;
260  overlayCam.enabled = false;
261  break;
262  }
263  }
264 
265  var tracker = SteamVR_Render.Top();
266  if (tracker != null)
267  scale = tracker.origin.localScale.x;
268  }
