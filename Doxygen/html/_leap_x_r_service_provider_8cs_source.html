235  void LateUpdate() {
236  var projectionMatrix = _cachedCamera.projectionMatrix;
237  switch (SystemInfo.graphicsDeviceType) {
238  #if !UNITY_2017_2_OR_NEWER
239  case UnityEngine.Rendering.GraphicsDeviceType.Direct3D9:
240  #endif
241  case UnityEngine.Rendering.GraphicsDeviceType.Direct3D11:
242  case UnityEngine.Rendering.GraphicsDeviceType.Direct3D12:
243  for (int i = 0; i < 4; i++) {
244  projectionMatrix[1, i] = -projectionMatrix[1, i];
245  }
246  // Scale and bias from OpenGL -> D3D depth range
247  for (int i = 0; i < 4; i++) {
248  projectionMatrix[2, i] = projectionMatrix[2, i] * 0.5f
249  + projectionMatrix[3, i] * 0.5f;
250  }
251  break;
252  }
253 
254  // Update Image Warping
255  Vector3 pastPosition; Quaternion pastRotation;
256  transformHistory.SampleTransform(imageTimeStamp
257  - (long)(warpingAdjustment * 1000f),
258  out pastPosition, out pastRotation);
259 
260  // Use _tweenImageWarping
261  var currCenterRotation = XRSupportUtil.GetXRNodeCenterEyeLocalRotation();
262 
263  var imageReferenceRotation = _temporalWarpingMode != TemporalWarpingMode.Off
264  ? pastRotation
265  : currCenterRotation;
266 
267  Quaternion imageQuatWarp = Quaternion.Inverse(currCenterRotation)
268  * imageReferenceRotation;
269  imageQuatWarp = Quaternion.Euler(imageQuatWarp.eulerAngles.x,
270  imageQuatWarp.eulerAngles.y,
271  -imageQuatWarp.eulerAngles.z);
272  Matrix4x4 imageMatWarp = projectionMatrix
273  * Matrix4x4.TRS(Vector3.zero, imageQuatWarp, Vector3.one)
274  * projectionMatrix.inverse;
275  Shader.SetGlobalMatrix("_LeapGlobalWarpedOffset", imageMatWarp);

355  protected virtual LeapTransform GetWarpedMatrix(long timestamp,
356  bool updateTemporalCompensation = true) {
357  LeapTransform leapTransform;
358 
359  //Calculate a Temporally Warped Pose
360  if (Application.isPlaying
361  && updateTemporalCompensation
362  && transformHistory.history.IsFull
363  && _temporalWarpingMode != TemporalWarpingMode.Off) {
364  transformHistory.SampleTransform(timestamp
365  - (long)(warpingAdjustment * 1000f)
366  - (_temporalWarpingMode ==
367  TemporalWarpingMode.Images ? -20000 : 0),
368  out warpedPosition, out warpedRotation);
369  }
370 
371  //Calculate the Current Pose
372  Pose currentPose = Pose.identity;
373  if (_deviceOffsetMode == DeviceOffsetMode.Transform && deviceOrigin != null) {
374  currentPose.position = deviceOrigin.position;
375  currentPose.rotation = deviceOrigin.rotation;
376  } else if (!Application.isPlaying) {
377  currentPose.position = currentPose.rotation * Vector3.up * deviceOffsetYAxis
378  + currentPose.rotation * Vector3.forward * deviceOffsetZAxis;
379  currentPose.rotation = Quaternion.Euler(deviceTiltXAxis, 0f, 0f);
380  currentPose = transform.ToLocalPose().Then(currentPose);
381  } else {
382  transformHistory.SampleTransform(timestamp, out currentPose.position,
383  out currentPose.rotation);
384  }
385 
386  //Choose between Warped and Current Pose
387  bool useCurrentPosition = (_temporalWarpingMode == TemporalWarpingMode.Off) || !Application.isPlaying;
388  warpedPosition = useCurrentPosition ? currentPose.position : warpedPosition;
389  warpedRotation = useCurrentPosition ? currentPose.rotation : warpedRotation;
390 
391  //Apply offsets (when applicable)
392  if (Application.isPlaying) {
393  if (_deviceOffsetMode != DeviceOffsetMode.Transform) {
394  warpedPosition += warpedRotation * Vector3.up * deviceOffsetYAxis
395  + warpedRotation * Vector3.forward * deviceOffsetZAxis;
396  warpedRotation *= Quaternion.Euler(deviceTiltXAxis, 0f, 0f);
397  }
398 
399  warpedRotation *= Quaternion.Euler(-90f, 180f, 0f);
400  }
401 
402  if (transform.parent != null && _deviceOffsetMode != DeviceOffsetMode.Transform) {
403  leapTransform = new LeapTransform(
404  transform.parent.TransformPoint(warpedPosition).ToVector(),
405  (transform.parent.rotation * warpedRotation).ToLeapQuaternion(),
406  transform.lossyScale.ToVector() * 1e-3f);
407  } else {
408  leapTransform = new LeapTransform(warpedPosition.ToVector(),
409  warpedRotation.ToLeapQuaternion(),
410  transform.lossyScale.ToVector() * 1e-3f);
411  }
412 
413  leapTransform.MirrorZ();
414 
415  return leapTransform;
416  }

426  protected void OnPreCullHandTransforms(Camera camera) {
427  if (updateHandInPrecull) {
428  //Don't update pre cull for preview, reflection, or scene view cameras
429  switch (camera.cameraType) {
430  case CameraType.Preview:
431  #if UNITY_2017_1_OR_NEWER
432  case CameraType.Reflection:
433  #endif
434  case CameraType.SceneView:
435  return;
436  }
437 
438  if (Application.isPlaying
439  && !manualUpdateHasBeenCalledSinceUpdate
440  && _leapController != null) {
441  manualUpdateHasBeenCalledSinceUpdate = true;
442 
443  //Find the left and/or right hand(s) to latch
444  Hand leftHand = null, rightHand = null;
445  LeapTransform precullLeftHand = LeapTransform.Identity;
446  LeapTransform precullRightHand = LeapTransform.Identity;
447  for (int i = 0; i < CurrentFrame.Hands.Count; i++) {
448  Hand updateHand = CurrentFrame.Hands[i];
449  if (updateHand.IsLeft && leftHand == null) {
450  leftHand = updateHand;
451  } else if (updateHand.IsRight && rightHand == null) {
452  rightHand = updateHand;
453  }
454  }
455 
456  //Determine their new Transforms
457  var interpolationTime = CalculateInterpolationTime();
458  _leapController.GetInterpolatedLeftRightTransform(
459  interpolationTime + (ExtrapolationAmount * 1000),
460  interpolationTime - (BounceAmount * 1000),
461  (leftHand != null ? leftHand.Id : 0),
462  (rightHand != null ? rightHand.Id : 0),
463  out precullLeftHand,
464  out precullRightHand);
465  bool leftValid = precullLeftHand.translation != Vector.Zero;
466  bool rightValid = precullRightHand.translation != Vector.Zero;
467  transformHands(ref precullLeftHand, ref precullRightHand);
468 
469  //Calculate the delta Transforms
470  if (rightHand != null && rightValid) {
471  _transformArray[0] =
472  Matrix4x4.TRS(precullRightHand.translation.ToVector3(),
473  precullRightHand.rotation.ToQuaternion(),
474  Vector3.one)
475  * Matrix4x4.Inverse(Matrix4x4.TRS(rightHand.PalmPosition.ToVector3(),
476  rightHand.Rotation.ToQuaternion(),
477  Vector3.one));
478  }
479  if (leftHand != null && leftValid) {
480  _transformArray[1] =
481  Matrix4x4.TRS(precullLeftHand.translation.ToVector3(),
482  precullLeftHand.rotation.ToQuaternion(),
483  Vector3.one)
484  * Matrix4x4.Inverse(Matrix4x4.TRS(leftHand.PalmPosition.ToVector3(),
485  leftHand.Rotation.ToQuaternion(),
486  Vector3.one));
487  }
488 
489  //Apply inside of the vertex shader
490  Shader.SetGlobalMatrixArray(HAND_ARRAY_GLOBAL_NAME, _transformArray);
491  }
492  }
493  }
